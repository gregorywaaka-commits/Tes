#  HOW THIS MOD'S ON-ACTION SYSTEM WORKS:
#  ─────────────────────────────────────
#  The mod drives gameplay through three CK3 pulse hooks:
#    • on_yearly_pulse  — fires once per year per living ruler;
#                         used for flavour, rare events, and
#                         long-period checks (plagues, Thu'um, etc.)
#    • on_monthly_pulse — fires every month; used for spread/
#                         decay ticks (flu, Kamal freeze).
#    • on_county_occupied — fires when any county changes hands;
#                           used for Kamal ice-curse.
#    • on_character_death  — fires on death; used for cleanup.
#
#  The random_events weight table controls probability:
#    The no-op '0' entry carries the bulk of the weight (200)
#    so that most years nothing happens — keeping events rare
#    and meaningful.  The other entries are weights relative
#    to that pool.  For example weight 5 means roughly 1-in-41
#    chance vs. the no-op in any given year for an eligible ruler.
#

#  frequently and can generate a lot of event triggers.

# ---------------------------------------------------------------
#  ON CHARACTER DEATH
#  Fires when any character dies.
#  Used to clean up trait flags that should not persist after
#  death.  Note: CK3 character variables/flags are automatically
#  destroyed with the character, but some global_vars may need
#  cleanup if they reference a specific character.
# ---------------------------------------------------------------
on_character_death = {
    events = {
        kothringi.060    # Removes knahaten_immune flag bookkeeping if the immune character was the last
        # Thu'um flags: character_flags die with the character — no manual cleanup needed.
        # Dragon soul counter (dragon_souls_absorbed_count) is a character variable — auto-cleared on death.
        # Greybeard_rank and thuumvoi_rank variables are also character-scoped — auto-cleared.
    }
    effect = {
        # --- Tower Stone / Amulet cleanup (pre-existing) ---
        if = {
            limit = { has_trait = tower_stone_keeper }
            every_held_county = {
                limit = { has_county_variable = tower_stone_present }
                remove_county_variable = tower_stone_present
            }
        }
        if = {
            limit = { has_character_flag = amulet_of_kings_claimed }
            remove_character_flag = amulet_of_kings_claimed
        }
        # --- Silvenar / Green Lady death link ---
        # If the Silvenar dies, the Green Lady dies too (and vice-versa).
        if = {
            limit = { has_trait = bosmer_silvenar }
            every_player = {
                limit = {
                    has_trait = green_lady_bound
                    has_character_flag = silvenar_green_lady_paired
                }
                kill_character = yes
            }
        }
        if = {
            limit = { has_trait = green_lady_bound }
            every_player = {
                limit = {
                    has_trait = bosmer_silvenar
                    has_character_flag = silvenar_green_lady_paired
                }
                kill_character = yes
            }
        }
    }
}
