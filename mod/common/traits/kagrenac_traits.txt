# ---------------------------------------------------------------
#  NEREVARINE VICTOR
#  The Nerevarine who completed the ancient prophecy of Azura:
#  retrieved the Tools of Kagrenac, descended into Red Mountain,
#  and struck the Heart of Lorkhan to end Dagoth Ur's divine
#  corruption.  An epoch-defining act.  The Blight is gone.
#  The Tribunal's immortality now slowly fades without the Heart.
#  Gained via kagrenac.030 only.
# ---------------------------------------------------------------
nerevarine_victor = {
    category = commander

    martial     = 5
    learning    = 3
    diplomacy   = 2
    intrigue    = 2
    prowess     = 5

    monthly_piety_gain_mult    = 0.30    # Azura's prophecy fulfilled — a living saint
    monthly_prestige_gain_mult = 0.30    # The most celebrated figure in Dunmer history
    health = 1                            # Completing the prophecy lifts a spiritual weight

    command_modifier = {
        advantage  = 5
        damage     = 0.15               # The Nerevarine strikes with the weight of destiny
        max_morale = 0.15
    }

    flag = nerevarine_victor
}

# ---------------------------------------------------------------
#  ALMALEXIA'S HAND
#  A warrior-priest personally chosen and invested by Almalexia
#  herself — one of the elite Hand who carry her divine authority
#  across Morrowind.  The goddess's blessing sharpens their blade
#  and their diplomacy alike: they speak with her voice and fight
#  with her strength.
#  Gained via pledge_as_almalexias_hand decision and
#  almalexia_hands.000 investiture event.
# ---------------------------------------------------------------
almalexias_hand = {
    category = lifestyle

    martial   = 3
    prowess   = 3
    diplomacy = 2

    monthly_piety_gain_mult    = 0.20    # Directly invested by Almalexia — highest Tribunal honour
    monthly_prestige_gain_mult = 0.15    # The Hands are known and respected across Morrowind

    command_modifier = {
        advantage  = 3
        max_morale = 0.10    # Almalexia's blessing sustains the faithful in extremis
    }

    flag = almalexias_hand
}
