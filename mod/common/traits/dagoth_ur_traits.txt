#  TRAITS IN THIS FILE:
#  ────────────────────
#  sixth_house_cultist       — Secret follower of the Sixth House (House Dagoth).
#                              Hidden loyalties, strong intrigue, piety/prestige penalty.
#  corprus_afflicted         — Infected with Dagoth Ur's divine corprus disease.
#                              Severe penalties but immunity to other diseases (lore accurate).
#  ash_devotee               — Advanced Sixth House initiate undergoing Blight-touched
#                              transformation.  Locked to Sixth House cultists.
#  ghost_fence_keeper        — A Dunmer lord who commits forces to maintaining the Ghost
#                              Fence around Red Mountain.  Defensive and devotional bonuses.
#  nerevarine_marked         — Marked by Azura as a potential incarnation of the Nerevarine.

#  heart_scholar_ascendant   — Route A apex (Kagrenac's Ambition): taps Heart of Lorkhan

#                              and sustained by periodic return to the Heart.  Degrades in
#                              three phases when the Heart is cut off (2E 882) and destroyed.
#  heart_fused_ascendant     — Route B apex (Kagrenac's Ambition): direct fusion with the
#                              Heart of Lorkhan, Dagoth Ur-style.  Vastly powerful but
#                              inseparable from the Heart — instant death when it is
#                              destroyed at 3E 427.
#
#  LORE CONTEXT:
#  ─────────────
#  The Sixth House (House Dagoth) never truly died after the Battle of Red Mountain.
#  Voryn Dagoth kept the Sharmat's inner fire alive, slowly amassing sleeper agents
#  across Tamriel.  Those who hear the Dreamer's whisper become sixth_house_cultists;
#  those who go further embrace corprus transformation; the truly devoted become
#  ash_devotees touched by the Blight itself.  On the other side, ghost_fence_keepers
#  are the Dunmer lords who sacrifice resources to hold back the Blight, while the
#  nerevarine_marked is the prophesied hero who will end Dagoth Ur's threat.
#
#  EXTERNAL DEPENDENCIES:
#  ─────────────────────

# ---------------------------------------------------------------
#  CORPRUS AFFLICTED
#  A character infected with Dagoth Ur's divine disease — a
#  supernatural plague created by drawing on the Heart of Lorkhan.
#  Most who contract it die or become monsters.  In lore, corprus
#  provides total immunity to ALL other diseases (the body is
#  already entirely subsumed by Lorkhan's power).
# ---------------------------------------------------------------
corprus_afflicted = {
    category = health
    is_good   = no

    health       = -3
    prowess      = -4
    diplomacy    = -3
    learning     = -2
    stress_gain_mult = 0.30    # The disease eats at mind and body

    # Lore: corprus confers complete immunity to other diseases
    # (no clean mechanical equivalent in CK3 — represented by
    # the health penalty partially offset by disease resistance)

    flag = corprus_afflicted
}

# ---------------------------------------------------------------
#  ASH DEVOTEE
#  An advanced Sixth House initiate who has partly embraced
#  the Blight-touched transformation — skin dusted grey,
#  eyes lit by Dagoth Ur's dream-fire.  They are still
#  recognisably mortal, but barely.  Gained through prolonged
#  Sixth House devotion (see dagoth_ur.020).
# ---------------------------------------------------------------
ash_devotee = {
    category = health
    is_good   = no

    health         = -1
    intrigue       = 4
    learning       = 2           # Blight-visions grant strange insight
    prowess        = 3           # Dagoth Ur's power strengthens the body even as it twists it

    monthly_prestige_gain_mult = -0.20   # A visibly Blight-touched character is feared
    monthly_piety_gain_mult    = -0.30   # Viewed as a monster by the Tribunal and Divines

    flag = ash_devotee
}

# ---------------------------------------------------------------

#  to maintaining the Ghost Fence — the great Tribunal barrier
#  that holds the Blight within Red Mountain.  Defensive bonuses
#  reflect the hardened, faith-tempered warriors of the Fence.
# ---------------------------------------------------------------
ghost_fence_keeper = {
    category = lifestyle

    monthly_piety_gain_mult    = 0.15    # The Ghost Fence is an act of Tribunal devotion
    monthly_prestige_gain_mult = 0.10    # Respected by all Dunmer as a defender of Morrowind
    martial   = 2
    prowess   = 2
    command_modifier = {
        advantage  = 3
        max_morale = 0.10    # Defending home and faith inspires extraordinary courage
    }

    flag = ghost_fence_keeper
}

# ---------------------------------------------------------------
#  HEART SCHOLAR ASCENDANT  (Route A apex — Kagrenac's Ambition)
#  A ruler who has tapped the Heart of Lorkhan using Kagrenac's
#  Tools — the Tribunal method.  Divine power is borrowed and
#  sustained; the practitioner must periodically return to the
#  Heart to renew the connection.  The Tribunal held this status
#  for three thousand years.
#
#  Power degrades in three phases when access to the Heart is
#  severed (2E 882) and again when the Heart is destroyed (3E 427).
#  Survival is possible via the CHIM escape (see §19.10b).
# ---------------------------------------------------------------
heart_scholar_ascendant = {
    category = lifestyle
    is_good   = yes

    learning       = 8     # Proximity to a dead god's power sharpens everything
    diplomacy      = 4     # The Tribunal were rulers as much as gods
    martial        = -2    # They are no longer primarily warriors

    monthly_piety          = 5     # Objectively more divine than they were
    stress_gain_mult       = -0.20 # The Heart's vastness provides perspective
    monthly_prestige_gain_mult = 0.10

    flag = heart_scholar_ascendant
}

# ---------------------------------------------------------------
#  HEART FUSED ASCENDANT  (Route B apex — Kagrenac's Ambition)
#  A ruler who has directly fused with the Heart of Lorkhan,
#  Dagoth Ur's method — without the intermediary of the Tools.
#  The connection is total, not mediated.  The character carries
#  a fragment of the Heart's presence within them at all times.
#
#  They do not need to renew the connection.  They are part of it.
#  When the Heart is destroyed (3E 427), they die instantly —
#  there is nothing to decouple from.
# ---------------------------------------------------------------
heart_fused_ascendant = {
    category = lifestyle
    is_good   = yes

    learning       = -2    # Intellect partially overwritten by the merge
    intrigue       = 6     # The Heart sees through all pretence
    martial        = 6     # Lorkhan's power flows into combat
    prowess        = 4

    monthly_piety          = 8     # They ARE the power now
    stress_gain_mult       = -0.40 # The Heart's vastness absorbs mortal concerns
    dread_gain_mult        = 0.40  # The Blight emanation frightens those nearby

    flag = heart_fused_ascendant
}

    monthly_piety_gain_mult    = -0.10   # The Tribunal would call this heresy; so would most faiths

    flag = akullakhan_scholar
}
