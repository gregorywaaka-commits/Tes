# -------------------------------------------------------------
#  LORE BASIS:
#    - Echmer (Uutak Mythos): uplifted bat-folk created by the
#      Noraken Dwemer clan via tonal architecture (sound magic);
#      native to Yneslea.
#    - orgnum_blessed uses is_immortal = yes — correct per lore:
#      Orgnum the Maormer king is genuinely immortal.
#    - clavicus_bargain_made: the Stillrise Village Kothringi
#      bargained with Clavicus Vile to survive the Knahaten Flu
#      (2E 560) and were turned into undead; stress_gain_mult = 0.2
#      reflects the psychological weight of immortality-as-undead.
#    - Knahaten Flu (2E 560): wiped out the Kothringi; survivors
#      who escaped on the Crimson Ship were never seen again.
#    - Kamal hibernate through winter on Akavir ("Snow Hell") and
#      thaw to invade; kamal_thawed represents peak battle-readiness.
#    - Dreugh have a natural land/sea life-cycle phase; the two
#      phase traits are mutually exclusive via opposites.
# -------------------------------------------------------------
#  HOW TO ADD A NEW TRAIT:
#    1. Choose a unique snake_case key.
#    2. Assign category:
#         special   — racial/heritage markers
#         lifestyle — acquired or phase-based states
#         fame      — high-rank or legendary achievements
#    3. Place the trait in the correct race/culture section and
#       add a descriptive comment above the block.
#    4. Use only valid CK3 modifier keys (see SYNTAX RULES below).
#    5. Add  flag = <trait_key>  at the end of the block.
#    6. Create matching loc keys in:
#         localization/english/traits/
#         lore_races_traits_l_english.yml
#    7. Add a trait icon at:
#         gfx/interface/icons/traits/<trait_key>.dds
# -------------------------------------------------------------
#  CK3 SYNTAX RULES FOR THIS FILE:
#    monthly_piety              — flat monthly piety (± integer)
#    monthly_piety_gain_mult    — multiplier on piety gain
#                                 (use instead of "piety_mult")
#    monthly_prestige           — flat monthly prestige
#    monthly_prestige_gain_mult — multiplier on prestige gain
#                                 (use instead of "prestige_mult")
#    stress_gain_mult           — multiplier on stress gain;
#                                 negative = stress reduction
#                                 (use instead of "stress_loss_mult")
#    is_immortal = yes          — prevents natural death by age
#    can_have_children = no     — blocks heir generation
#    command_modifier { }       — active only while in command
#
#    INVALID KEYS — never use:
#      piety_mult, prestige_mult, stress_loss_mult
# =============================================================

# Bargained with Clavicus Vile for survival — the Stillrise Village fate.
# LORE NOTE: The Kothringi of Stillrise Village (Black Marsh, c. 2E 560) made
# a pact with Clavicus Vile to gain immunity to the Knahaten Flu.  Vile granted
# it — at the cost of their mortality.  They became permanently undead: immune
# to the Flu, immune to disease, immune to death by natural means, but cut off
# from the Divines and from any normal mortal life.  They are neither truly alive
# nor truly dead.  The stress_gain_mult reflects the psychological weight of
# existing as undead; the piety penalty reflects how faith communities
# (and the characters themselves) regard the Daedric bargain that made them so.
clavicus_bargain_made = {
    category = lifestyle
    monthly_piety    = -3   # faith communities distrust Daedric bargains
    monthly_prestige = 1
    stress_gain_mult = 0.2  # the price of immortality-as-undead weighs heavily
    flag = clavicus_bargain_made
}

# =============================================================
#  Keptu / Nedic Tribe Traits
#
#  LORE NOTE — KEPTU:
#  The Keptu were an ancient proto-human people of High Rock
#  (not Yokudan / Redguard ancestors — that is a common
#  conflation).  They predated Aldmeri settlement of Tamriel and
#  inhabited the Breton heartland.  Their smithing tradition
#  centred on the Bloodroot Forge — a Daedric artifact of
#  Hircine associated with bestial enhancements — as attested
#  in ESO: Horns of the Reach.  keptu_lineage represents a
#  character who traces direct descent from Keptu smithing
#  masters; the stewardship bonus reflects mastery of craft.
#
#  NOTE: The event file (minor_lore_races_events.txt) previously
#  described the Keptu as "proto-Redguard" — this is incorrect.
#  The Keptu are a distinct High Rock culture; the Redguard
#  (Yokudans) arrived on Tamriel c. 1E 808, millennia after
#  the Keptu's decline.
# =============================================================

# A ruler who traces direct Keptu lineage (Bloodroot Forge smiths of High Rock)

    category = special
    stewardship = 2
    monthly_prestige = 1
