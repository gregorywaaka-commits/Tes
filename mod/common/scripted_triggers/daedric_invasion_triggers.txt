# =============================================================
#  Daedric Invasion — Scripted Triggers
#  File: mod/common/scripted_triggers/daedric_invasion_triggers.txt
#
#  HOW THE DAEDRIC INVASION TRIGGER SYSTEM WORKS:
#  ─────────────────────────────────────────────────
#  These triggers are the "is X currently true?" checks for the
#  Daedric invasion system.  They are used by:
#    • Decision is_shown / is_valid blocks
#    • Event trigger / immediate blocks
#    • Other scripted triggers (composed checks)
#    • Effects (begin/escalate/end effects check these)
#
#  GLOBAL VARIABLE SUMMARY:
#  ─────────────────────────
#  global_var:active_daedric_prince — string key of the current Prince
#    template (e.g. "molag_bal_template").  EXISTS iff an invasion is active.
#  global_var:daedric_invasion_phase — integer 1/2/3.  Phase increases
#    over time via escalate_daedric_invasion.  Phase 3 = maximum threat.
#  global_var:daedric_invasion_started_days — integer days counter.
#    Incremented monthly by daedric_invasion.050.
#    escalate check fires when this >= 1825 (5 years).
#
#  HOW TO ADD A NEW PRINCE:
#  ─────────────────────────
#  1. Add is_X_invasion trigger here (copy pattern below).
#  2. Add the prince's template to character_templates/daedric_princes.txt.
#  3. Add an else_if block to begin_daedric_invasion (daedric_invasion_effects.txt).
#  4. Add cleanup to end_daedric_invasion (daedric_invasion_effects.txt).
#  5. Add a seek_X decision to daedric_prince_decisions.txt.
#  6. Add a daedric_champion.X.001 event chain to daedric_champion_events.txt.
#  7. Add the X_aspect and X_champion traits to daedric_invasion_traits.txt.
#  8. Add any unique modifiers to lore_races_modifiers.txt.
# =============================================================

is_daedric_invasion_active = {
    exists = global_var:active_daedric_prince
}

# True for the AI character who IS the Daedric Prince invader.
# is_ai = yes prevents this from ever being true for a player character.
is_daedric_prince_invader = {
    has_trait = daedric_prince    # the daedric_prince trait (daedric_invasion_traits.txt)
    is_ai = yes
}

# True when a specific county (province scope) has an Oblivion Gate.
# Gates are opened by open_oblivion_gate_effect and closed by close_oblivion_gate_effect.
# The variable is set on the county itself (not a character variable).
province_has_oblivion_gate = {
    has_variable = oblivion_gate_active    # set by open_oblivion_gate_effect in daedric_invasion_effects.txt
}

# True when any of the ruler's held counties has a Gate.
# Used to show/gate decisions that let the ruler personally deal with a nearby Gate.
ruler_threatened_by_oblivion_gate = {
    any_held_county = {
        has_variable = oblivion_gate_active
    }
}

# Can the invasion escalate to the next phase?
# Requirements:
#   • invasion phase variable exists (invasion is active)
#   • phase < 3 (cannot escalate beyond phase 3)
#   • 1825+ days have passed since start (~5 years per phase)
# The 5-year escalation period means a typical invasion lasts ~15 years
# from start to maximum threat — long enough to require sustained response.
daedric_invasion_can_escalate = {
    exists = global_var:daedric_invasion_phase
    global_var:daedric_invasion_phase < 3
    global_var:daedric_invasion_started_days >= 1825    # 5 years = 1825 days in CK3
}

# True once the invasion has reached maximum escalation (phase 3).
# At this point the invasion cannot get worse — only be ended.
daedric_invasion_at_full_power = {
    exists = global_var:daedric_invasion_phase
    global_var:daedric_invasion_phase >= 3
}

# Can a NEW invasion be triggered?
# Requirements:
#   • no invasion currently active (prevents two simultaneous invasions)
#   • date is past 2E.100 (early Interregnum guard — EK2 starts at 2E 440,
#     so this is effectively always true, but guards against edge cases)
#   • there exists at least one eligible large non-champion realm to target
# Note: The actual Prince selection and conditions are checked inside
#       daedric_invasion.000 (daedric_invasion_events.txt), not here.
daedric_invasion_trigger_conditions = {
    NOT = { is_daedric_invasion_active = yes }
    current_date >= 2E.100                   # minimum date — all invasions set in Interregnum
    any_ruler = {
        realm_size >= 30                     # only realms large enough to be worth targeting
        NOT = { has_trait = daedric_champion }    # champions are already serving a Prince
    }
}

# Is the character a champion or devotee of ANY Daedric Prince?
# Note: daedric_champion is the generic base trait (Mehrunes Dagon, Boethiah,
# Hircine, Namira, Molag Bal); the other Princes have unique champion traits.
# See daedric_invasion_traits.txt for full trait definitions.
is_any_daedric_champion = {
    OR = {
        has_trait = daedric_champion          # base: Dagon/Boethiah/Hircine/Namira/Molag Bal
        has_trait = azura_champion
        has_trait = mephala_champion
        has_trait = clavicus_vile_champion
        has_trait = vaermina_champion
        has_trait = sheogorath_champion
        has_trait = peryite_champion
        has_trait = sanguine_champion
        has_trait = meridia_champion
        has_trait = malacath_champion
        has_trait = nocturnal_champion
        has_trait = hermaeus_mora_champion
        has_trait = jyggalag_champion
    }
}

# Checks if a character actively worships or is aligned with Daedra.
# Used to block certain virtuous decisions (e.g. Amulet of Kings quest)
# from Daedra worshippers, and to gate Daedric champion quests.
# Three paths to being a Daedra worshipper:
#   1. Faith with doctrine_daedra_revered — formal religious worship
#   2. Faith with doctrine_daedra_accepted — their faith tolerates Daedra
#   3. daedric_vassal trait — submitted to a Prince during an invasion
#   4. Being any kind of champion — an active relationship with a Prince
is_daedra_worshipper = {
    OR = {
        faith = { has_doctrine = doctrine_daedra_revered }
        faith = { has_doctrine = doctrine_daedra_accepted }
        has_trait = daedric_vassal
        is_any_daedric_champion = yes
    }
}
