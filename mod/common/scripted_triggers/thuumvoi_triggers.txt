# =============================================================
#  Thu'um / Voice — Scripted Triggers
#  EK2 is required; 'dragonborn' references EK2's existing trait.
# =============================================================

# Is the character on the Voice path at any level?
is_on_voice_path = {
    OR = {
        has_trait = thu_um_initiate
        has_trait = thu_um_adept
        has_trait = thu_um_master
        has_trait = greybeard_aspirant
        has_trait = greybeard_of_high_hrothgar
        has_trait = first_voice_of_the_greybeards
    }
}

# Is the character an active member of the Greybeard Order?
is_greybeard = {
    OR = {
        has_trait = greybeard_aspirant
        has_trait = greybeard_of_high_hrothgar
        has_trait = first_voice_of_the_greybeards
    }
}

# Is the character eligible to begin Voice training?
# Requirements: old enough, not already trained, no conflicting path
can_learn_thu_um = {
    is_adult = yes
    NOT = { is_on_voice_path = yes }
    NOT = { has_character_flag = thuumvoi_quest_started }
    OR = {
        learning >= 8
        has_trait = dragonborn   # EK2 trait
    }
}

# Can the character advance their Voice training?
can_advance_thu_um = {
    is_adult = yes
    NOT = { has_modifier = thuumvoi_advancement_cooldown }
    OR = {
        has_trait = thu_um_initiate
        has_trait = thu_um_adept
    }
}

# Can the character seek the Greybeards?
can_seek_greybeards = {
    is_adult = yes
    NOT = { is_greybeard = yes }
    NOT = { has_character_flag = greybeard_quest_started }
    OR = {
        has_trait = thu_um_master   # Masters may seek the Order
        has_trait = dragonborn      # Dragonborn are always welcome
    }
}

# Is this character the EK2 Dragonborn?
is_dragonborn = {
    has_trait = dragonborn   # EK2-defined trait
}

# Can the Dragonborn harness their full power?
can_harness_dovah_blood = {
    is_dragonborn = yes
    NOT = { has_trait = dovahkiin_soul_awakened }
    NOT = { has_character_flag = dovahkiin_quest_started }
    is_adult = yes
}

# Is the character the First Voice?
is_first_voice = {
    has_trait = first_voice_of_the_greybeards
}

# Is a Greybeard quest ready to begin?
can_begin_greybeard_leadership = {
    has_trait = greybeard_of_high_hrothgar
    NOT = { has_character_flag = greybeard_lead_quest_1 }
    NOT = { is_first_voice = yes }
    NOT = { has_modifier = thuumvoi_advancement_cooldown }
}

# ---------------------------------------------------------------
#  DRAGONBORN SOUL / DOVAH NATURE TRIGGERS
# ---------------------------------------------------------------

# Has the Dragonborn absorbed multiple Dragon Souls (5+)?
has_multiple_dragon_souls = {
    has_trait = dovahkiin_soul_awakened
    check_variable = { name = dragon_souls_absorbed_count value >= 5 }
}

# Has the Dragonborn absorbed many Dragon Souls (10+)?
has_many_dragon_souls = {
    has_trait = dovahkiin_soul_awakened
    check_variable = { name = dragon_souls_absorbed_count value >= 10 }
}

# Is the Dovah nature at risk of taking over?
# (10+ souls but not yet resolved)
dovah_nature_at_risk = {
    has_many_dragon_souls = yes
    NOT = { has_trait = dovah_nature_resisted }
    NOT = { has_trait = dovah_nature_dominant }
}

# Can the Dragonborn learn from a Word Wall?
# Word Walls open instantly to a Dragonborn — they need no teacher.
# Cooldown applies to prevent rapid stacking.
can_learn_from_word_wall = {
    is_dragonborn = yes
    is_adult = yes
    NOT = { has_modifier = dovahkiin_word_wall_cooldown }
    # Must have at least begun their awakening (or have basic training)
    OR = {
        has_trait = dovahkiin_soul_awakened
        has_trait = dragon_souls_growing
        has_trait = dragon_souls_brimming
        has_trait = thu_um_initiate     # Even an initiate Dragonborn can read a Word Wall
    }
}

# Can the Dragonborn seek out and absorb another Dragon Soul?
# (Represents seeking a dragon to slay)
can_absorb_another_soul = {
    has_trait = dovahkiin_soul_awakened
    is_adult = yes
    NOT = { has_modifier = absorb_soul_cooldown }
    NOT = { has_trait = dovah_nature_dominant }     # Dominated Dragonborn can't focus
}

# ---------------------------------------------------------------
#  AMULET OF KINGS TRIGGERS
# ---------------------------------------------------------------

# Does this character hold the Amulet of Kings?
has_amulet_of_kings = {
    has_character_flag = amulet_of_kings_bearer
}

# Can the character seek the Amulet of Kings?
# Must be Dragonborn — only those of Dragonblood can fulfil the Covenant
can_seek_amulet_of_kings = {
    is_dragonborn = yes
    is_adult = yes
    NOT = { has_amulet_of_kings = yes }
    NOT = { has_character_flag = amulet_quest_started }
    piety >= 500
}

# Can the character attempt to relight the Dragonfires?
# Requires the Amulet, the Dragonborn blood, and the White-Gold Tower
can_relight_dragonfires = {
    has_amulet_of_kings = yes
    is_dragonborn = yes
    NOT = { has_character_flag = dragonfires_lit }
    # Must control or have access to the White-Gold Tower (Cyrodiil)
    OR = {
        any_held_county = { this = title:c_imperial_city }
        any_realm_county = { this = title:c_imperial_city }
    }
}
