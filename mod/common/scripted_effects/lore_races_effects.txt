# ---- Echmer: Establish First Contact ----
establish_echmer_contact = {
    set_global_variable = { name = echmer_contact_established value = yes }
    # Spawn an Echmer scholar envoy at the player's court
    create_character = {
        template = echmer_scholar_template
        location = root.capital_county
        save_scope_as = echmer_envoy
    }
    scope:echmer_envoy = {
        set_relation_friend = { target = root reason = reason_echmer_envoy }
    }
    every_ruler = {
        limit = { is_alive = yes }
        trigger_event = { id = echmer.001 }
    }
}

# ---- Maormer: Maormer Raid on Coastal County ----
maormer_raid_effect = {
    # Apply a raided modifier to the target county
    add_province_modifier = {
        modifier = maormer_raided_modifier
        days = 730
    }
    # Grant the county owner a decision opportunity to retaliate
    scope:raided_county_owner = {
        trigger_event = { id = maormer.001 }
    }
}

# ---- Maormer: Tame a Sea Serpent ----
tame_sea_serpent_effect = {
    root = {
        add_trait = sea_serpent_master
        add_prestige = 1000
        add_piety    = 500
        trigger_event = { id = maormer.010 }
    }
}

# ---- Sload: Unleash the Thrassian Plague ----
unleash_thrassian_plague = {
    set_global_variable = { name = thrassian_plague_active value = yes }
    set_global_variable = { name = thrassian_plague_origin value = root }
    # Apply province plague modifier to 5 random coastal counties
    every_coastal_county = {
        limit = {
            NOT = { holder = { is_sload = yes } }
        }
        add_province_modifier = {
            modifier = thrassian_plague_modifier
            days = -1
        }
    }
    every_ruler = {
        limit = { is_alive = yes }
        trigger_event = { id = sload.001 }
    }
}

# ---- Sload: Form the All-Flags Navy (response to Thrassian Plague) ----
form_all_flags_navy_effect = {
    set_global_variable = { name = all_flags_navy_active value = yes }
    # Notify all coastal rulers
    every_ruler = {
        limit = {
            is_alive = yes
            any_held_county = { is_coastal = yes }
        }
        trigger_event = { id = sload.010 }
    }
}

# ---- Sload: Destroy Thras (end the plague) ----
destroy_thras_effect = {
    remove_global_variable = thrassian_plague_active
    remove_global_variable = thrassian_plague_origin
    remove_global_variable = all_flags_navy_active
    # Remove plague modifiers from all counties
    every_county = {
        limit = { has_modifier = thrassian_plague_modifier }
        remove_province_modifier = thrassian_plague_modifier
    }
    every_ruler = {
        limit = { is_alive = yes }
        trigger_event = { id = sload.020 }
    }
}

# ---- Kothringi: Knahaten Flu Outbreak ----
trigger_knahaten_flu = {
    set_global_variable = { name = knahaten_flu_active value = yes }
    # Infect counties in Black Marsh first
    every_county = {
        limit = {
            title_province = { geographical_region = world_black_marsh }
        }
        add_province_modifier = {
            modifier = knahaten_flu_modifier
            days = -1
        }
    }
    every_ruler = {
        limit = {
            is_alive = yes
            primary_title = { capital_county = { geographical_region = world_black_marsh } }
        }
        trigger_event = { id = kothringi.001 }
    }
}

# ---- Kamal: Begin Summer Invasion ----
begin_kamal_invasion = {
    set_global_variable = { name = kamal_invasion_active value = yes }
    create_character = {
        template = dir_kamal_template
        location = title:d_kamal_lands
        save_scope_as = dir_kamal_char
    }
    scope:dir_kamal_char = {
        create_army = {
            name = "Kamal War Host"
            men_at_arms = { type = kamal_warriors  stack = 600 }
            men_at_arms = { type = kamal_ice_mages stack = 100 }
        }
    }
    every_ruler = {
        limit = { is_alive = yes }
        trigger_event = { id = kamal.001 }
    }
}

# ---- Kamal: End Invasion (repelled or winter) ----
end_kamal_invasion = {
    remove_global_variable = kamal_invasion_active
    scope:dir_kamal_char = {
        death = { death_reason = death_banished }
    }
    every_ruler = {
        limit = { is_alive = yes }
        trigger_event = { id = kamal.020 }
    }
}

# ---- Elder Towers: Activate a Tower Stone ----
activate_tower_stone_effect = {
    # scope:tower_county must be set before calling
    scope:tower_county = {
        set_variable = { name = tower_stone_present value = yes }
        add_province_modifier = {
            modifier = tower_stone_active_modifier
            days = -1
        }
    }
    # Increment the global active tower count
    if = {
        limit = { NOT = { exists = global_var:active_tower_count } }
        set_global_variable = { name = active_tower_count value = 1 }
    }
    else = {
        change_global_variable = { name = active_tower_count add = 1 }
    }
    root = {
        add_trait    = tower_stone_keeper
        add_prestige = 1500
        add_piety    = 500
    }
}

# ---- Elder Towers: Deactivate / Destroy a Tower Stone ----
deactivate_tower_stone_effect = {
    scope:tower_county = {
        remove_variable = tower_stone_present
        remove_province_modifier = tower_stone_active_modifier
        add_province_modifier = {
            modifier = tower_fallen_modifier
            days = 3650
        }
    }
    change_global_variable = { name = active_tower_count subtract = 1 }
    # Reality begins to fray if too few Towers remain
    if = {
        limit = { world_stability_critical = yes }
        every_ruler = {
            limit = { is_alive = yes }
            trigger_event = { id = towers.015 }
        }
    }
}
