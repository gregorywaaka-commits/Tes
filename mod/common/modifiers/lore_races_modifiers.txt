# =============================================================
#  Province / Character Modifiers — All Systems
#  File: mod/common/modifiers/lore_races_modifiers.txt
#
#  !! IMPORTANT FOR FUTURE AGENTS !!
#  ALL modifiers for this mod live in this single file regardless
#  of which system they belong to (Daedric, Thu'um, Guild, etc.).
#  Do NOT create a second lore_races_modifiers.txt — it would
#  shadow this one.  Append new modifiers at the end.
#
#  MODIFIER TYPES IN CK3:
#  ─────────────────────
#  • Character modifiers: affect a character's stats/income.
#    Applied with: add_modifier = { modifier = X days = N }
#    days = -1 means PERMANENT until manually removed.
#  • Province modifiers: affect a specific county's development,
#    income, supply.
#    Applied with: add_province_modifier = { modifier = X days = N }
#  • Empty modifiers { }: act as "cooldown flags" — they apply
#    no stats but their presence can be checked with has_modifier.
#    These are used to gate decisions and prevent event spam.
#
#  VALUE SCALE REFERENCE (for future contributors):
#  ─────────────────────────────────────────────────
#  monthly_income = 0.1        ≈ +10% income (small)
#  monthly_income = -0.4       ≈ -40% income (severe plague)
#  development_growth_factor = -0.5  ≈ halved growth (devastating)
#  supply_limit_mult = -0.3    ≈ armies need more time to feed (military impact)
#  monthly_prestige_gain_mult  = 0.1 ≈ +10% prestige gain (small)
#  dread_gain_mult = 0.25      ≈ dread increases 25% faster
#  prowess = 4                 ≈ +4 points to personal combat (notable)
#  diplomacy = 3               ≈ +3 to all diplomacy skill checks
#  command_modifier.advantage = 5   ≈ significant tactical edge in battles
#  stress_gain_mult = 0.2      ≈ stressful situations 20% more stressful
#
#  HOW MODIFIERS ARE APPLIED/REMOVED (cross-reference):
#  ──────────────────────────────────────────────────────
#  See scripted_effects/ files for where each modifier is
#  applied and removed — search for 'add_modifier' and
#  'remove_modifier' calls with the modifier name.
# =============================================================
