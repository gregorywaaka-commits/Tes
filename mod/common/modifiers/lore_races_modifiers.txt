#  Province / Character Modifiers — All Systems

#
#  !! IMPORTANT FOR FUTURE AGENTS !!
#  ALL modifiers for this mod live in this single file regardless
#  of which system they belong to (Daedric, Thu'um, Guild, etc.).
#  Do NOT create a second lore_races_modifiers.txt — it would
#  shadow this one.  Append new modifiers at the end.
#
#  MODIFIER TYPES IN CK3:
#  ─────────────────────
#  • Character modifiers: affect a character's stats/income.
#    Applied with: add_modifier = { modifier = X days = N }
#    days = -1 means PERMANENT until manually removed.
#  • Province modifiers: affect a specific county's development,
#    income, supply.
#    Applied with: add_province_modifier = { modifier = X days = N }
#  • Empty modifiers { }: act as "cooldown flags" — they apply
#    no stats but their presence can be checked with has_modifier.
#    These are used to gate decisions and prevent event spam.
#
#  VALUE SCALE REFERENCE (for future contributors):
#  ─────────────────────────────────────────────────
#  monthly_income = 0.1        ≈ +10% income (small)
#  monthly_income = -0.4       ≈ -40% income (severe plague)
#  development_growth_factor = -0.5  ≈ halved growth (devastating)
#  supply_limit_mult = -0.3    ≈ armies need more time to feed (military impact)
#  monthly_prestige_gain_mult  = 0.1 ≈ +10% prestige gain (small)
#  dread_gain_mult = 0.25      ≈ dread increases 25% faster
#  prowess = 4                 ≈ +4 points to personal combat (notable)
#  diplomacy = 3               ≈ +3 to all diplomacy skill checks
#  command_modifier.advantage = 5   ≈ significant tactical edge in battles
#  stress_gain_mult = 0.2      ≈ stressful situations 20% more stressful
#
#  HOW MODIFIERS ARE APPLIED/REMOVED (cross-reference):
#  ──────────────────────────────────────────────────────
#  See scripted_effects/ files for where each modifier is
#  applied and removed — search for 'add_modifier' and
#  'remove_modifier' calls with the modifier name.
# =============================================================

maormer_raided_modifier = {
    monthly_income    = -0.2    # Raided merchants and warehouses
    supply_limit_mult = -0.1    # Disrupted grain stores
}

# Applied when a peace/truce has been negotiated with Maormer.
# Acts as a cooldown flag — no stat effect, just prevents raid events.
# Duration: set by decision outcome (typically 1825 days = 5 years).
maormer_truce_modifier = {
    build_speed_mult = 0.0   # no-op; presence of modifier prevents maormer_raid_possible
}

# Applied to coastal counties caught in a Maormer tempest.
# Worse than a raid — sustained weather disruption rather than a single attack.
maormer_tempest_modifier = {
    monthly_income    = -0.15
    supply_limit_mult = -0.2
}

# Bonus for a county that repelled a Maormer raid — local morale is high.
# Duration: 365 days (1 year).
repelled_maormer_modifier = {
    monthly_prestige_gain_mult = 0.1   # County holder gains prestige for successful defence
}

# Applied to a ruler who has personal hatred for Maormer (lost family/land to raids).
maormer_hatred_modifier = {
    levy_size = 0.1                     # Subjects rally against the hated enemy
    monthly_prestige_gain_mult = 0.05
}

# ---------------------------------------------------------------
#  SLOAD (Thrassian Slug-People)
#  Context: Sload are slug-like necromancers from the coral
#  island of Thras.  They unleashed the Thrassian Plague in
#  1E 2260.  In 2E their schemes are subtler.
#  Applied by: unleash_thrassian_plague (lore_races_effects.txt)
#  Removed by: destroy_thras_effect (lore_races_effects.txt)
#  Checked by: thrassian_plague_active trigger
# ---------------------------------------------------------------

# Applied to every coastal county while the Thrassian Plague is active.
# The plague was the most devastating in Tamrielic history — values reflect that.
# Duration: permanent until destroy_thras_effect runs (days = -1).
thrassian_plague_modifier = {
    monthly_income            = -0.4    # Trade collapses; ports abandon commerce
    supply_limit_mult         = -0.3    # Dead farmers = less food
    development_growth_factor = -0.5    # No-one is building while dying
}

# Applied to counties that quarantine themselves against the Plague.
# Lesser impact than the plague, but costs them too — quarantines hurt trade.
# Duration: 730 days (2 years).
quarantine_modifier = {
    monthly_income    = -0.1
    supply_limit_mult = -0.1
}

}

# ---------------------------------------------------------------
#  ELDER TOWERS (World Anchors)
#  Context: The Towers (White-Gold, Adamantine, Red Mountain,
#  Coral, Orichalc) are massive structures that anchor Mundus
#  (the mortal plane) in reality.  If enough fall, the world
#  becomes unstable.
#  Applied by: activate_tower_stone_effect, deactivate_tower_stone_effect
#  Checked by: tower_X_active triggers, world_stability_critical trigger
# ---------------------------------------------------------------

# Applied to a county that holds an active Tower Stone.
# The Stone radiates stability and power into the surrounding land.
# Duration: permanent (days = -1) until the Stone is deactivated.
tower_stone_active_modifier = {
    monthly_income            = 0.2     # Stone draws pilgrims and trade
    development_growth_factor = 0.3     # Land flourishes under the Tower's influence
    supply_limit_mult         = 0.15    # Stable reality = reliable supply
}

# Applied when a Tower Stone is destroyed or stolen.
# The land weakens — reality is less stable here.
# Duration: 3650 days (10 years).
tower_fallen_modifier = {
    monthly_income            = -0.2
    development_growth_factor = -0.2
}

# Applied to the ruler who controls the White-Gold Tower (Imperial City).
# White-Gold is Nirn's Tower — the most powerful of all the Towers.
# Duration: permanent while they hold the Imperial City (applied by towers.002 event).
white_gold_keeper_modifier = {
    monthly_prestige_gain_mult = 0.3    # Controlling the heart of the Empire lends enormous prestige
    monthly_income             = 0.2    # The City is the wealthiest in Tamriel
    diplomacy                  = 3      # The Tower's Covenant gives authority
}

# Applied to the ruler who controls Red Mountain (Morrowind).
# Red Mountain contains the Heart of Lorkhan — the dead god's power source.
# Duration: permanent while they control Red Mountain (applied by towers.004 event).
red_mountain_keeper_modifier = {
    monthly_piety_gain_mult = 0.3   # Closeness to Lorkhan's Heart is spiritually profound
    martial                 = 3     # The volcanic landscape forges hard soldiers
    prowess                 = 4     # Red Mountain's power bleeds into the keeper
}

# Applied to ALL rulers when active_tower_count drops to 1 or below.
# Reality is fracturing — the world can feel it.
# Duration: 1825 days (5 years) per application.
reality_fracturing_modifier = {
    monthly_prestige_gain_mult = -0.2   # The world is ending; no-one cares about court politics
    development_growth_factor  = -0.5   # Development halts as reality becomes unreliable
    monthly_income             = -0.3   # Trade collapses when the world may end any day
}

# Empty cooldown — prevents a ruler from sabotaging Towers repeatedly.
# Duration: 3650 days (10 years) on success, 1825 days on failure.
tower_sabotage_cooldown = { }
