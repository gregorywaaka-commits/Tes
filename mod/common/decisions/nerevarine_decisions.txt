# ---------------------------------------------------------------
#  CLAIM THE NEREVARINE PROPHECY
#  When the prophecy's chosen vessel has fallen without completing
#  their task, Azura calls for a new incarnation.  Any ruler who
#  bears the azura_champion mark and has the will to carry the
#  burden may step forward to take up the mantle.
#
#  LORE NOTE — RACE-AGNOSTIC PROPHECY:
#  The Nerevarine is a reincarnation of Lord Nerevar's SOUL, not a
#  product of Dunmeri blood.  Azura's prophecy speaks of incarnation
#  across lifetimes — the soul returns wearing whatever flesh it
#  chooses.  In the canonical events of TES III, the Nerevarine is
#  an outlander (the player can be any race), explicitly called
#  "the Stranger" in some Ashlander prophecy variants.  The souls
#  of past Nervarines remain in Azura's realm (Moonshadow) and can
#  be encountered in Morrowind — proving the spirit of Nerevar does
#  not extinguish between lives but persists as multiple echoes.
#  Race does not gatekeep the prophecy; Azura's recognition does.
#  [SOURCE: TES III:Morrowind — Nerevarine Prophecies; Azura shrine
#  dialogue; UESP Lore:Nerevarine]
#
#  MULTI-SPIRIT MECHANIC:
#  The active_nerevarine_spirit_count global var tracks simultaneous
#  living vessels.  A second claimant while a first lives becomes
#  a "second spirit" — lore-valid, as Nerevar's soul is large enough
#  to manifest more than once.  nerevarine.310 fires for all new
#  claimants; nerevarine.300 fires for non-Dunmer specifically.
#
#  The new Nerevarine inherits the quest at its current stage —
#  if the previous vessel had already been named Hortator or
#  recognised by the Ashlanders, that recognition transfers.
#
#  Note: if false_nerevarine_count >= 3, the piety cost increases
#  significantly — Azura is harder to approach after repeated failures.
# ---------------------------------------------------------------
claim_nerevarine_prophecy = {
    picture = "gfx/interface/icons/decisions/decision_realm.dds"
    desc = claim_nerevarine_prophecy_desc

    is_shown = {
        # Prophecy can be claimed when the previous vessel has died OR a second
        # spirit can rise alongside an existing one (multi-spirit lore).
        OR = {
            has_global_flag = nerevarine_quest_incomplete
            # A second soul claiming while first still lives — second spirit mechanic
            AND = {
                any_living_character = { has_trait = nerevarine_marked }
                global_var:active_nerevarine_spirit_count >= 1
                global_var:active_nerevarine_spirit_count < 3   # lore cap: three past souls visible in Morrowind
            }
        }
        has_trait = azura_champion
        # Race is NOT a requirement — the Nerevarine is a soul reincarnation,
        # not a racial prophecy.  Any worthy vessel may be called.
        is_ruler = yes
        NOT = { has_trait = nerevarine_marked }
        NOT = { has_trait = nerevarine_victor }
        NOT = { has_trait = nerevarine_forsaken }
        dagoth_ur_awakened = yes
        NOT = { has_global_flag = dagoth_ur_defeated }
    }

    is_valid = {
        piety >= 300
        prestige >= 200
        is_adult = yes
    }

    is_valid_showing_failures_only = {
        is_alive = yes
    }

    effect = {
        add_piety    = -300
        add_prestige = -200
        add_trait    = nerevarine_marked
        # Clear the "quest open" flag only when no other spirit is active
        if = {
            limit = { has_global_flag = nerevarine_quest_incomplete }
            remove_global_flag = nerevarine_quest_incomplete
        }
        # Track simultaneous spirits
        if = {
            limit = { NOT = { exists = global_var:active_nerevarine_spirit_count } }
            set_global_variable = { name = active_nerevarine_spirit_count value = 0 }
        }
        change_global_variable = { name = active_nerevarine_spirit_count add = 1 }
        if = {
            limit = { NOT = { exists = global_var:false_nerevarine_count } }
            set_global_variable = { name = false_nerevarine_count value = 0 }
        }
        add_character_modifier = {
            modifier = dagoth_ur_champion_modifier
            days = 3650
        }
        add_prestige = 400
        # Outlander Nerevarine revelation — fires for non-Dunmer claimants
        if = {
            limit = { NOT = { has_culture = dunmer_culture } }
            trigger_event = { id = nerevarine.300 days = 3 }
        }
        # Vision of past incarnations — fires for all new claimants
        trigger_event = { id = nerevarine.310 days = 7 }
        custom_tooltip = nerevarine_claim_done_tt
    }

    ai_will_do = {
        base = 10
        modifier = {
            add = 20
            has_trait = brave
        }
        modifier = {
            add = 10
            has_trait = ambitious
        }
    }
}


# ---------------------------------------------------------------
#  BETRAY AZURA FOR DAGOTH
#  At the Heart's Chamber — all three Tools gathered, the moment
#  of the prophecy at hand — the Nerevarine chooses the Sharmat
#  over Azura.  This requires deep Sixth House sympathy or a cunning
#  contempt for prophetic obligation.
#
#  Azura's champion status is revoked.  The nerevarine_forsaken
#  trait replaces it.  The character joins the Dagoth route (Route B
#  of Path F: heart_fused_ascendant) and a world event notifies all
#  Dunmer rulers of the prophecy's betrayal.
#
#  Note: this is a COMPLEMENTARY decision alongside kagrenac.030 option B.
#  Both produce the same outcome.  This decision is the explicit decision-panel
#  version — it requires having all three Tool flags, meaning the character
#  went through kagrenac.010/015/020 and reached the threshold.  Unlike
#  the kagrenac.030 event (which fires automatically 30 days after kagrenac.020),
#  this decision is player-driven and allows deliberate timing of the betrayal.
# ---------------------------------------------------------------
betray_azura_for_dagoth = {
    picture = "gfx/interface/icons/decisions/decision_personal.dds"
    desc = betray_azura_for_dagoth_desc

    is_shown = {
        has_trait = nerevarine_marked
        has_character_flag = sunder_found
        has_character_flag = keening_found
        has_character_flag = wraithguard_found
        dagoth_ur_awakened = yes
        NOT = { has_trait = nerevarine_victor }
        NOT = { has_character_flag = nerevarine_turned_dagoth }
        NOT = { has_character_flag = betray_dagoth_pending }
        NOT = { has_global_flag = dagoth_ur_defeated }
    }

    is_valid = {
        OR = {
            has_trait = sixth_house_cultist
            has_trait = ash_devotee
            intrigue >= 16
        }
    }

    is_valid_showing_failures_only = {
        is_adult = yes
        is_alive = yes
    }

    effect = {
        # Remove prophecy connection to Azura
        remove_trait = nerevarine_marked
        if = {
            limit = { has_trait = azura_champion }
            remove_trait = azura_champion
        }
        # Mark as forsaken and turned
        add_trait = nerevarine_forsaken
        set_character_flag = nerevarine_turned_dagoth
        add_character_modifier = {
            modifier = nerevarine_turned_modifier
            days = 1825    # 5-year world reaction
        }
        # Join the Dagoth fusion path
        if = {
            limit = { NOT = { has_trait = heart_fused_ascendant } }
            add_trait = heart_fused_ascendant
        }
        # Lose the piety accumulated as Azura's champion
        add_piety    = -500
        add_prestige = 200    # The Sixth House hails the turned champion
        add_stress   = -20    # The burden of the prophecy is lifted, replaced by the dream
        # Clear the tools (the moment has passed)
        remove_character_flag = kagrenac_quest_started
        remove_character_flag = sunder_found
        remove_character_flag = keening_found
        remove_character_flag = wraithguard_found
        # Notify all Dunmer rulers
        every_ruler = {
            limit = {
                is_alive = yes
                has_culture = dunmer_culture
            }
            trigger_event = { id = nerevarine.200 days = 7 }
        }
        custom_tooltip = nerevarine_turned_tt
    }

    ai_will_do = {
        base = 2    # AI very rarely takes this path
        modifier = {
            add = 15
            has_trait = sixth_house_cultist
            has_trait = ash_devotee
        }
        modifier = {
            add = 8
            has_trait = ambitious
            intrigue >= 16
        }
        modifier = {
            add = -20
            has_trait = devoted    # Deeply devoted characters will not betray their patron
        }
    }
}

# ---------------------------------------------------------------
#  SLAY THE NEREVARINE
#  A ruler tapping the Heart of Lorkhan — whether as a Tribunal-
#  method scholar or as a Dagoth-method fused character — has every
#  reason to prevent the Nerevarine from severing the Heart's binding.
#  The sever would mean instant death for fused rulers, or the end
#  of accumulated divine power for scholars.
#
#  This is a personal and terrible choice.  The player targets the
#  living Nerevarine-marked character and launches an assassination
#  attempt.  Success requires martial or intrigue strength.
#  Failure injures the attacker and destroys prestige.
#
#  NOTE: Killing the Nerevarine only delays the Heart's departure.
#  The world flag nerevarine_quest_incomplete is set, allowing
#  another vessel to claim the prophecy and resume.
# ---------------------------------------------------------------
slay_the_nerevarine = {
    picture = "gfx/interface/icons/decisions/decision_personal.dds"
    desc = slay_the_nerevarine_desc

    is_shown = {
        OR = {
            has_trait = heart_scholar_ascendant
            has_trait = heart_fused_ascendant
        }
        is_ruler = yes
        dagoth_ur_awakened = yes
        NOT = { has_global_flag = dagoth_ur_defeated }
        NOT = { has_character_flag = nerevarine_slay_attempted }
        # Must be an active Nerevarine to target
        any_living_character = {
            has_trait = nerevarine_marked
            NOT = { has_trait = nerevarine_victor }
        }
    }

    is_valid = {
        prestige >= 500
        OR = {
            martial  >= 12
            intrigue >= 12
        }
    }

    is_valid_showing_failures_only = {
        is_adult = yes
        is_alive = yes
    }

    effect = {
        add_prestige = -500
        set_character_flag = { flag = nerevarine_slay_attempted days = 3650 }
        trigger_event = { id = nerevarine.030 days = 7 }
    }

    ai_will_do = {
        base = 5
        modifier = {
            add = 30
            has_trait = heart_fused_ascendant    # Fused rulers will die if Heart is severed — strong motivation
        }
        modifier = {
            add = 15
            has_trait = heart_scholar_ascendant
            martial >= 14
        }
        modifier = {
            add = 20
            has_trait = selfish
        }
    }
}
