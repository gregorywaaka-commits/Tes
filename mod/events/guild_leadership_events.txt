#  Namespace: guild_leader
#
#  Each guild has a 3-event story chain + a climactic leadership
#  event.  Events are triggered by the quest-start decision and
#  fire sequentially via character flags.
#
#  Flag convention:
#    guild_X_quest_1    — first chapter seen
#    guild_X_quest_2    — second chapter seen
#    guild_X_quest_done — questline complete; claim decision unlocked
# =============================================================

# ================================================================
#  MAGES GUILD  — "The Stolen Flame"
#  Trigger: at Wizard rank, "Begin Leadership Quest" decision
#  Story:   A forbidden Daedric study is spreading inside the
#           Guild; the Arch-Magister is hiding it to save face.
#           You must expose the rot and take control.
# ================================================================

# Chapter 1 — A junior mage shows you anomalous readings in the
# library vault.  Something is very wrong.
guild_leader.mages.001 = {
    type = character_event
    title = guild_leader.mages.001.t
    desc = guild_leader.mages.001.desc
    theme = realm

    trigger = {
        is_mages_guild_member = yes
        has_trait = mages_guild_wizard
        NOT = { has_character_flag = guild_mages_quest_1 }
    }

    option = {
        name = guild_leader.mages.001.a  # "Investigate further — discreetly."
        add_character_flag = guild_mages_quest_1
        add_stress = 5
        trigger_event = { id = guild_leader.mages.002 days = 30 }
    }
    option = {
        name = guild_leader.mages.001.b  # "Report to the Arch-Magister at once."
        add_character_flag = guild_mages_quest_1
        add_prestige = -50    # The Arch-Magister thanks you — and buries the report
        add_modifier = { modifier = guild_recently_rejected days = 365 }
        # Quest is delayed but not ended
        trigger_event = { id = guild_leader.mages.002 days = 90 }
    }
}

# Chapter 2 — The forbidden study is confirmed: Daedric soul-trap
# experiments on unwilling subjects.  The Arch-Magister is implicated.
guild_leader.mages.002 = {

    title = guild_leader.mages.002.t

    theme = realm

    trigger = {
        has_character_flag = guild_mages_quest_1
        NOT = { has_character_flag = guild_mages_quest_2 }
    }

        name = guild_leader.mages.002.a  # "Gather the evidence.  We take this to the Council."
        add_character_flag = guild_mages_quest_2
        add_learning = 1
        trigger_event = { id = guild_leader.mages.003 days = 30 }
    }
    option = {
        name = guild_leader.mages.002.b  # "Destroy the evidence — the Guild's reputation must come first."
        add_character_flag = guild_mages_quest_2
        add_prestige = -100   # Reputation hit for suppressing the truth
        add_piety    = -50    # Modest spiritual cost; greater than Fighters Guild case due to Daedric nature
        trigger_event = { id = guild_leader.mages.003 days = 30 }
    }
}

# Leadership Event — The Council names you Arch-Magister
guild_leader.mages.lead = {
    type = character_event
    title = guild_leader.mages.lead.t
    desc = guild_leader.mages.lead.desc
    theme = realm

    trigger = {
        has_character_flag = guild_mages_quest_done
        NOT = { has_trait = mages_guild_archmagister }
    }

    option = {
        name = guild_leader.mages.lead.a  # "I accept the burden of Arch-Magister."
        remove_trait    = mages_guild_wizard
        add_trait       = mages_guild_archmagister
        change_variable = { name = guild_mages_rank add = 1 }
        add_modifier = { modifier = guild_leader_mages_modifier days = -1 }
        add_prestige = 600
        add_piety    = 200
        remove_character_flag = guild_mages_quest_done
        add_character_flag    = guild_mages_is_leader
    }
}
