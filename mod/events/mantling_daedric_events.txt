mantling_daedric.azura_rank3 = {
    type = character_event
    title = mantling_daedric.azura_rank3.t
    desc  = mantling_daedric.azura_rank3.desc
    theme = intrigue

    option = {
        name = mantling_daedric.azura_rank3.a   # "The distinction matters less and less."
        add_prestige = 175
        add_piety = 75
        # The 25% possession check fires in the apotheosis event immediate block
    }
}

mantling_daedric.azura_apotheosis = {
    type = character_event
    title = mantling_daedric.azura_apotheosis.t
    desc  = mantling_daedric.azura_apotheosis.desc
    theme = intrigue

    trigger = {
        has_character_flag = ww_mantling_azura_active
        var:ww_mantling_azura_rank >= 4
        NOT = { has_character_flag = ww_mantling_azura_completed }
    }

    immediate = {
        # 25% possession risk fires before trait grant
        random = {
            chance = 25
            trigger_event = { id = mantling_daedric.azura_possession days = 3 }
            set_character_flag = ww_daedric_mantle_failed
        }
        if = {
            limit = { NOT = { has_character_flag = ww_daedric_mantle_failed } }
            complete_path_mantling_azura = yes
        }
    }

    option = {
        name = mantling_daedric.azura_apotheosis.a
        trigger = { NOT = { has_character_flag = ww_daedric_mantle_failed } }
        add_prestige = 300
        add_piety = 200
    }

    option = {
        name = mantling_daedric.azura_apotheosis.uncertain
        trigger = { has_character_flag = ww_daedric_mantle_failed }
        add_stress = 30
        # Possession event already queued
    }
}

mantling_daedric.azura_possession = {
    type = character_event
    title = mantling_daedric.azura_possession.t
    desc  = mantling_daedric.azura_possession.desc
    theme = intrigue

    trigger = {
        has_character_flag = ww_daedric_mantle_failed
    }

    immediate = {
        # Become Azura's champion instead of her aspect
        remove_trait = walking_ways_seeker
        clear_all_ww_active_flags = yes
        remove_variable = ww_mantling_azura_rank
        remove_variable = ww_mantling_azura_milestones
        remove_character_flag = ww_daedric_mantle_failed
        add_trait = daedric_champion
    }

    option = {
        name = mantling_daedric.azura_possession.a
        add_stress = 30
        add_prestige = 100   # A champion of Azura still has power — just not the kind sought
    }
}
