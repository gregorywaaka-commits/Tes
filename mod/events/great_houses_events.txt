# =============================================================
#  Great Houses of Morrowind
#  Namespace: great_houses
#
#  LORE:
#  The five Great Houses of Morrowind are the political bedrock
#  of Dunmer civilisation — each house a distinct ideology, a
#  distinct philosophy of power:
#
#    Redoran   — the warrior house; honour, duty, and martial
#                excellence above all.  They guard Morrowind's
#                borders and hold the border wars as sacred.
#    Telvanni  — isolationist wizard-lords who grow their towers
#                from living mushroom-trees; they answer to no
#                Tribunal authority and no temporal law.
#    Indoril   — the Tribunal clergy's house; deeply pious,
#                holding the Temple above all secular concerns.
#    Hlaalu    — merchants and political operators; they survive
#                by making themselves indispensable to every
#                power that rises, and they always survive.
#    Dres      — lords of the Deshaan Plains, bordering Black
#                Marsh; they have built their wealth on the
#                slave trade, raiding Hist-bonded Argonians for
#                generations.
#
#  SYSTEM OVERVIEW:
#  ─────────────────
#  • great_houses.000  — Hortator candidacy: Great House council
#                        approaches a worthy Dunmer ruler
#  • great_houses.001  — Redoran Test: martial honour & combat
#  • great_houses.002  — Telvanni Negotiation: power & knowledge
#  • great_houses.003  — Indoril Piety: genuine faith & devotion
#  • great_houses.004  — Hlaalu Bargain: commerce & obligation
#  • great_houses.005  — Dres Compliance: slavery & conscience

#

# ---------------------------------------------------------------
# great_houses.000  A Great House Council Approaches
# Yearly flavour: a council of Great House lords rides to a
# worthy Dunmer ruler's hall with a proposition.  The five
# houses are seeking a Hortator — a war-leader to stand against
# a great threat — and this ruler's name has been spoken aloud
# at the council table.  The candidacy is an honour.  It is
# also a burden with no clear bottom.
# ---------------------------------------------------------------
great_houses.000 = {
    type = character_event
    title = great_houses.000.t
    desc = great_houses.000.desc
    theme = diplomacy

    trigger = {
        is_ruler = yes
        has_culture = dunmer_culture
        NOT = { has_trait = morrowind_hortator }
        NOT = { has_character_flag = hortator_redoran }
        NOT = { has_character_flag = hortator_selection_active }
    }

    option = {
        name = great_houses.000.a   # "I accept the honour. Tell me what the houses require."
        add_character_flag = { flag = hortator_selection_active days = 730 }
        add_prestige = 100
        add_stress   = 5
        trigger_event = { id = great_houses.001 days = 14 }
    }

    option = {
        name = great_houses.000.b   # "I am not ready. Let the council find another name."
        add_prestige = -50
        add_stress   = 5
    }

    option = {
        name = great_houses.000.c   # "An honour worth recognising — in gold as well as words."
        trigger = {
            gold >= 0   # always available; gold option
        }
        add_character_flag = { flag = hortator_selection_active days = 730 }
        add_gold     = 200
        add_prestige = 50
        add_stress   = 10
        trigger_event = { id = great_houses.001 days = 14 }
    }
}

# ---------------------------------------------------------------
# great_houses.001  The Redoran Test — Honour and the Spear
# House Redoran will not name a Hortator who cannot prove
# honour in combat.  They are warriors first; the council
# demands demonstration before they commit the house's name
# to any cause.  Duel their champion, present a battle record,
# or fall back on lineage.
# ---------------------------------------------------------------
great_houses.001 = {
    type = character_event
    title = great_houses.001.t
    desc = great_houses.001.desc
    theme = martial

    option = {
        name = great_houses.001.a   # "I will face your champion personally."
        if = {
            limit = { martial >= 12 }
            add_character_modifier = {
                modifier = hortator_redoran_support
                days = 1825
            }
            add_character_flag = { flag = hortator_redoran }
            add_prestige = 150
            trigger_event = { id = great_houses.002 days = 21 }
        }
        else = {
            add_stress   = 20
            add_prestige = -75
            custom_tooltip = great_houses_redoran_duel_failed_tt
        }
    }

    option = {
        name = great_houses.001.b   # "Let my battle record speak. I have not avoided war."
        if = {
            limit = { martial >= 10 }
            add_character_flag = { flag = hortator_redoran }
            add_prestige = 100
            trigger_event = { id = great_houses.002 days = 21 }
        }
        else = {
            add_stress   = 15
            add_prestige = -50
            custom_tooltip = great_houses_redoran_record_failed_tt
        }
    }

    option = {
        name = great_houses.001.c   # "My lineage speaks for itself. Redoran knows my blood."
        add_character_flag = { flag = hortator_redoran }
        add_stress   = 5
        add_prestige = 50
        trigger_event = { id = great_houses.002 days = 21 }
    }
}

# ---------------------------------------------------------------
# great_houses.005  Dres Compliance — The Slave-Lords of Deshaan
# House Dres is the last negotiation in the Hortator chain — and
# the most difficult.  Their power comes from the slave trade,
# and they do not welcome a war-leader who might disrupt it.
# The candidate must choose: accept the institution, condemn it,
# or find the narrow path of compromise that satisfies neither
# conscience nor council.
# ---------------------------------------------------------------
great_houses.005 = {
    type = character_event
    title = great_houses.005.t
    desc = great_houses.005.desc
    theme = intrigue

    option = {
        name = great_houses.005.a   # "The slave trade is a Morrowind institution. I will not interfere with it."
        add_character_modifier = {
            modifier = dres_slave_tribute
            days = 730
        }
        add_character_modifier = {
            modifier = hortator_dres_compliance
            days = 730
        }
        add_character_flag = { flag = hortator_dres }
        add_piety  = -150
        add_stress = 20
        if = {
            limit = {
                has_character_flag = hortator_redoran
                has_character_flag = hortator_telvanni
                has_character_flag = hortator_indoril
                has_character_flag = hortator_hlaalu
            }
            trigger_event = { id = great_houses.006 days = 30 }
        }
    }

    option = {
        name = great_houses.005.b   # "I will not endorse slavery. Condemn me for it if you must."
        add_piety    = 200
        add_prestige = -100
        add_stress   = 10
        custom_tooltip = great_houses_dres_condemned_tt
        if = {
            limit = {
                has_character_flag = hortator_redoran
                has_character_flag = hortator_telvanni
                has_character_flag = hortator_indoril
                has_character_flag = hortator_hlaalu
            }
            add_character_flag = { flag = hortator_dres }
            trigger_event = { id = great_houses.006 days = 30 }
        }
        else = {
            add_character_flag = { flag = hortator_dres }
        }
    }

    option = {
        name = great_houses.005.c   # "There is a middle path here. Let us find it together."
        if = {
            limit = { stewardship >= 12 }
            add_character_flag = { flag = hortator_dres }
            add_prestige = 50
            add_piety    = 50
            add_stress   = 5
            if = {
                limit = {
                    has_character_flag = hortator_redoran

            }
        }

            add_prestige = -50

        }
    }
}

    title = great_houses.022.t
    desc = great_houses.022.desc
    theme = martial

    trigger = {
        is_ruler = yes
        has_character_flag = dres_trade_engaged
        has_character_flag = hist_warning_received
        NOT = { has_character_flag = argonian_uprising_resolved }
    }

    option = {
        name = great_houses.022.a   # "Crush the uprising. Morrowind's borders must hold."

        add_prestige = 100
        add_piety    = -100
        add_stress   = 25

        custom_tooltip = great_houses_uprising_crushed_tt
    }

        }

            modifier = argonian_uprising_aftermath
            days = 1825
        }
        add_gold     = -300
        add_piety    = 100
        add_stress   = 10
        remove_character_flag = dres_trade_engaged
    }

    option = {
        name = great_houses.022.c   # "Withdraw. Let the border counties sort themselves."

            modifier = argonian_uprising_aftermath

    }
