mantling_jyggalag.apotheosis = {
    type = character_event
    title = mantling_jyggalag.apotheosis.t
    desc  = mantling_jyggalag.apotheosis.desc
    theme = intrigue

    trigger = {
        has_character_flag = ww_mantling_jyggalag_active
        var:ww_mantling_jyggalag_rank >= 4
        NOT = { has_character_flag = ww_mantling_jyggalag_completed }
        NOT = { has_character_flag = ww_daedric_mantle_failed }
    }

    immediate = {
        complete_path_mantling_jyggalag = yes
    }

    option = {
        name = mantling_jyggalag.apotheosis.a   # "The domain knows me. I walk where Jyggalag walked, and the pattern holds."
        add_prestige = 400
        add_piety = 150
        add_stress = -20
    }

    option = {
        name = mantling_jyggalag.apotheosis.b   # "Order without the Prince. The domain persists. I am its new instrument."
        add_prestige = 350
        add_dread = 15
    }
}

mantling_jyggalag.order_shattered = {
    type = character_event
    title = mantling_jyggalag.order_shattered.t
    desc  = mantling_jyggalag.order_shattered.desc
    theme = intrigue

    trigger = {
        has_character_flag = ww_daedric_mantle_failed
        has_character_flag = ww_mantling_jyggalag_active
    }

    immediate = {
        remove_trait = walking_ways_seeker
        clear_all_ww_active_flags = yes
        remove_variable = ww_mantling_jyggalag_rank
        remove_variable = ww_mantling_jyggalag_milestones
        remove_character_flag = ww_daedric_mantle_failed
        add_trait = obsessed
        add_character_modifier = {
            modifier = order_shattered_compulsion
            years = 25
        }
    }

    option = {
        name = mantling_jyggalag.order_shattered.a   # "The Order was too rigid. It broke me before I could break into it."
        add_stress = 35
        add_prestige = 50
    }
}
