# =============================================================
#  Thu'um / Voice Events
#  File: mod/events/thuumvoi_events.txt
#  Three namespaces (all declared below):
#    thuumvoi  — General Voice training path (thuumvoi.001–thuumvoi.010+)
#    greybeard — The Order of the Greybeards + leadership (greybeard.001–greybeard.lead.003)
#    dovahkiin — Dragonborn awakening chain (dovahkiin.001+)
#
#  WHAT THIS FILE DOES:
#  Implements the Thu'um (Voice) progression system, the
#  Greybeard Order quest chain, and the Dragonborn awakening
#  sequence.  Three namespace sections are maintained in one
#  file for cohesion; each namespace block is clearly marked.
#
#  EVENT INDEX — thuumvoi namespace:
#  thuumvoi.001 — The Voice Stirs (first training)
#  thuumvoi.002 — A Word of Power learned
#  thuumvoi.003 — The Second Shout (Initiate → Adept)
#  thuumvoi.004 — The Tongue (Adept → Master)
#  thuumvoi.005 — The Thu'um in Battle (FUS RO DAH / YOL TOOR SHUL / WULD)
#  thuumvoi.010 — The Fifth Soul (dragon_souls_growing: 5 souls absorbed)
#
#  EVENT INDEX — greybeard namespace:
#  greybeard.001      — The 7,000 Steps Begin
#  greybeard.002      — The Gates of High Hrothgar
#  greybeard.003      — Meeting the Greybeards (Arngeir speaks)
#  greybeard.004      — Learning the First Word Under Arngeir's Eye
#  greybeard.005      — The Throat of the World / Paarthurnax audience
#  greybeard.lead.001 — Leadership quest step 1
#  greybeard.lead.002 — Leadership quest step 2
#  greybeard.lead.003 — The Way Made Whole (leadership conclusion)
#  greybeard.post_001 — Post-leadership recurring event
#
#  EVENT INDEX — dovahkiin namespace:
#  dovahkiin.001 — The Blood Awakens (fired by awaken_dovah_blood effect)
#
#  EK2 IDENTIFIERS USED:
#  Scripted effects:   begin_voice_training, awaken_dovah_blood,
#                      advance_greybeard_rank
#  Traits:             thu_um_initiate, thu_um_adept, thu_um_master,
#                      greybeard_aspirant, greybeard_of_high_hrothgar,
#                      first_voice_of_the_greybeards, dovahkiin_soul_awakened,
#                      dragon_souls_growing
#  Modifiers:          thu_um_shout_cooldown, greybeard_leader_modifier
#  Variables:          greybeard_rank (character scope: change_variable)
#
#  LORE BASIS:
#  Lore timeframe: 2E 440–896 (the Interregnum).
#  The Greybeards are an ancient order on High Hrothgar founded
#  by Jurgen Windcaller after the Battle of Red Mountain (1E 416),
#  who advocate using the Voice only to honour the gods.
#  PAARTHURNAX: The great dragon at the summit of the Throat of the
#  World is Paarthurnax — Alduin's former lieutenant who turned
#  against him and has lived at the Throat since the Merethic Era.
#  He pre-dates the Greybeards: they built High Hrothgar beneath
#  the mountain he already inhabited, and he taught Jurgen Windcaller
#  himself.  His presence in the 2E timeframe is fully canonical —
#  unlike Arngeir (see below), no anachronism is involved.
#  ARNGEIR ANACHRONISM NOTE: Arngeir is the Greybeards' spokesman
#  in The Elder Scrolls V: Skyrim (4E 201).  His presence here in
#  the 2E Interregnum is a deliberate creative choice, representing
#  a long-lived or identically-named Greybeard elder.  This is
#  flagged as a known anachronism and may be replaced with a
#  generic Greybeard spokesman in a future revision.
#  Sources: TES V: Skyrim, Songs of the Return, The Voice of the
#  Sky, Jurgen Windcaller and His Epitaph.
#
#  HOW TO ADD A NEW EVENT:
#  1. Add <namespace>.NNN = { ... } under the appropriate
#     namespace section header in this file
#  2. Declare the namespace with namespace = X before the first
#     event of that block (the three declarations are at lines
#     ~12, ~177, ~567 — do not add a fourth)
#  3. Add localization keys to mod/localization/english/
#     thuumvoi_l_english.yml
# =============================================================

# ----------------------------------------------------------------
# thuumvoi.001  — The Voice Stirs  (first training event)
# Fired by begin_voice_training effect / seek_thu_um_training decision
# ----------------------------------------------------------------
thuumvoi.001 = {
    type  = character_event
    title = thuumvoi.001.t
    desc  = thuumvoi.001.desc
    theme = realm

    trigger = {
        has_character_flag = thuumvoi_quest_started
        has_trait = thu_um_initiate
        NOT = { has_character_flag = thuumvoi_event_1_seen }
    }

    option = {
        name = thuumvoi.001.a   # "Breathe.  Listen.  Let the world's voice speak to yours."
        add_character_flag = thuumvoi_event_1_seen
        add_learning = 1
        add_stress   = -5
    }
    option = {
        name = thuumvoi.001.b   # "What does this mean?  What have I tapped into?"
        add_character_flag = thuumvoi_event_1_seen
        add_piety  = 50
        add_stress = 10         # The power is terrifying
    }
}

# ----------------------------------------------------------------
# greybeard.lead  — Named First Voice
# The Order formally recognises you as their First Voice.
# ----------------------------------------------------------------
greybeard.lead = {
    type  = character_event
    title = greybeard.lead.t
    desc  = greybeard.lead.desc
    theme = realm

    trigger = {
        has_character_flag = greybeard_lead_quest_done
        NOT = { has_trait = first_voice_of_the_greybeards }
    }

    option = {
        name = greybeard.lead.a   # "I accept.  Speak rarely.  Speak truly.  Speak with the whole weight of the world."
        remove_trait    = greybeard_of_high_hrothgar
        add_trait       = first_voice_of_the_greybeards
        # greybeard_rank was initialised to 1 in join_greybeard_order,
        # incremented to 2 in initiate_greybeard; this brings it to 3 (max).
        if = {
            limit = { check_variable = { name = greybeard_rank value >= 2 } }
            change_variable = { name = greybeard_rank add = 1 }
        }
        else = {
            set_variable = { name = greybeard_rank value = 3 }  # safety fallback
        }
        add_modifier = { modifier = first_voice_modifier days = -1 }
        add_piety    = 1000
        add_prestige = 800
        remove_character_flag = greybeard_lead_quest_done
        add_character_flag    = greybeard_is_first_voice
    }
}

# The accumulated Dragon Souls press against the mortal self.
# Choose: resist or submit.
# ----------------------------------------------------------------

    type  = character_event
    title = thuumvoi.012.t
    desc  = thuumvoi.012.desc
    theme = realm

    trigger = {
        has_character_flag = dovah_crisis_pending
        NOT = { has_character_flag = dovah_crisis_resolved }
    }

    option = {
        name = thuumvoi.012.a   # "I am not a dragon.  I am Dovahkiin.  I choose."
        # Attempt to resist — requires piety; succeeds at cost
        add_character_flag = dovah_crisis_resolved
        remove_character_flag = dovah_crisis_pending
        if = {
            limit = { piety >= 300 }
            resist_dovah_nature = yes
        }
        else = {
            # Not enough spiritual grounding — the nature takes hold temporarily
            submit_to_dovah_nature = yes
        }
    }
    option = {
        name = thuumvoi.012.b   # "Let it come.  What mortal can stand against me now?"
        add_character_flag = dovah_crisis_resolved
        remove_character_flag = dovah_crisis_pending
        submit_to_dovah_nature = yes
    }
}
