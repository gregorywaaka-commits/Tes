# =============================================================
#  Daedric Invasion — Main Events
#  File: mod/events/daedric_invasion_events.txt
#  Namespace: daedric_invasion
#
#  WHAT THIS FILE DOES:
#  Core machinery for Daedric invasions of Tamriel.  Handles
#  invasion start/escalation/end, ruler notifications, rally
#  mechanics, submission, the dramatic banishment duel, and
#  cleanup on character death.
#
#  EVENT INDEX:
#  daedric_invasion.000 — Yearly pulse: attempt to start a new invasion (world_event)
#  daedric_invasion.001 — Invasion declared: notification to all rulers
#  daedric_invasion.002 — War declaration to defenders
#  daedric_invasion.005 — Invasion escalation notification
#  daedric_invasion.020 — Invasion ended — victory celebration
#  daedric_invasion.030 — A ruler rallies vassals (fires on vassals)
#  daedric_invasion.040 — A ruler submits to the Daedric Prince
#  daedric_invasion.045 — Player attempts to banish the Prince (dramatic duel)
#  daedric_invasion.050 — Monthly tick: increment timer, check escalation (hidden world_event)
#  daedric_invasion.060 — On-death cleanup: remove daedric_champion / daedric_vassal traits
#
#  EK2 IDENTIFIERS USED:
#  Scripted triggers:  daedric_invasion_trigger_conditions, is_daedric_invasion_active,
#                      daedric_invasion_can_escalate
#  Scripted effects:   begin_daedric_invasion, end_daedric_invasion,
#                      escalate_daedric_invasion
#  Global variables:   chosen_prince_key, active_daedric_prince,
#                      daedric_invasion_started_days, daedric_invasion_phase
#  Traits:             daedric_champion, daedric_vassal
#  Opinions:           rallied_together_opinion, ignored_call_opinion,
#                      submitted_to_daedra_opinion
#  Modifier:           wounded
#
#  LORE BASIS:
#  Daedric invasions are a recurring threat in the Second Era.
#  Mehrunes Dagon's invasions (most frequent), Molag Bal's Planemeld
#  (2E 578–582), and the Greymarch of Jyggalag (near 3E 400) are
#  the most significant.  Invasion weights reflect lore frequency.
#  Jyggalag is locked behind current_date >= 3E.400.
#
#  HOW TO ADD A NEW EVENT:
#  1. Add daedric_invasion.NNN = { ... } following the pattern below
#  2. Add localization keys to mod/localization/english/
#     daedric_invasion_l_english.yml
#  3. Wire into the invasion lifecycle via begin_daedric_invasion,
#     escalate_daedric_invasion, or end_daedric_invasion scripted effects
# =============================================================

namespace = daedric_invasion

# ------------------------------------------------------------
# daedric_invasion.000
# Yearly pulse check: attempt to start a new Daedric invasion.
# ------------------------------------------------------------
daedric_invasion.000 = {
    type = world_event
    title = daedric_invasion.000.t
    desc = daedric_invasion.000.desc
    theme = war

    trigger = {
        daedric_invasion_trigger_conditions = yes
    }

    immediate = {
        # Randomly pick which Daedric Prince leads this invasion
        # Weights reflect lore frequency: Dagon most common; Jyggalag rarest (the Greymarch)
        random_list = {
            25 = { set_global_variable = { name = chosen_prince_key value = "mehrunes_dagon_template" } }
            15 = { set_global_variable = { name = chosen_prince_key value = "molag_bal_template" } }
            10 = { set_global_variable = { name = chosen_prince_key value = "boethiah_template" } }
            10 = { set_global_variable = { name = chosen_prince_key value = "hircine_template" } }
             5 = { set_global_variable = { name = chosen_prince_key value = "namira_template" } }
             # LORE NOTE: Azura's 'invasion' is a prophetic visitation — her per-Prince events
             # (daedric_prince_invasion.azura.*) are wholly beneficial: prophecy, Azura's Star,
             # and guidance against other Daedric armies.  The hostile CB is a mod abstraction
             # to integrate her into the invasion framework; it represents political upheaval
             # caused by her followers rather than a literal military invasion.
             # Weight lowered from 5 → 2 to reflect her non-hostile nature in lore.
             2 = { set_global_variable = { name = chosen_prince_key value = "azura_template" } }
             8 = { set_global_variable = { name = chosen_prince_key value = "mephala_template" } }
             4 = { set_global_variable = { name = chosen_prince_key value = "clavicus_vile_template" } }
             5 = { set_global_variable = { name = chosen_prince_key value = "vaermina_template" } }
             3 = { set_global_variable = { name = chosen_prince_key value = "sheogorath_template" } }
             5 = { set_global_variable = { name = chosen_prince_key value = "peryite_template" } }
             2 = { set_global_variable = { name = chosen_prince_key value = "sanguine_template" } }
             # LORE NOTE: Meridia is Aedric-adjacent and despises undead; her 'invasion' events
             # (daedric_prince_invasion.meridia.*) grant Dawnbreaker and purge necromancers — net
             # beneficial.  The CB is a mod abstraction for the disruption her zealous followers
             # cause.  Weight lowered from 3 → 1 to reflect her exceptionally rare hostility.
             1 = { set_global_variable = { name = chosen_prince_key value = "meridia_template" } }
             5 = { set_global_variable = { name = chosen_prince_key value = "malacath_template" } }
             3 = { set_global_variable = { name = chosen_prince_key value = "nocturnal_template" } }
             2 = { set_global_variable = { name = chosen_prince_key value = "hermaeus_mora_template" } }
             2 = {
                 # Jyggalag is trapped inside Sheogorath throughout the 2E Interregnum —
                 # his true form cannot manifest until near 3E 400 (the Greymarch cycle).
                 # Base weight 2 is fully cancelled by the modifier below before that date,
                 # giving a net weight of 0 in the 2E timeframe.
                 modifier = {
                     add = -2   # net = 0 before 3E 400; Jyggalag cannot appear in the 2E
                     NOT = { current_date >= 3E.400 }
                 }
                 set_global_variable = { name = chosen_prince_key value = "jyggalag_template" }
             }
        }
        begin_daedric_invasion = yes
    }

    option = {
        name = daedric_invasion.000.a
        custom_tooltip = daedric_invasion_started_tt
    }
}

# ------------------------------------------------------------
# daedric_invasion.001
# Notification to all rulers that an invasion has begun.
# ------------------------------------------------------------
daedric_invasion.001 = {
    type = character_event
    title = daedric_invasion.001.t
    desc = {
        first_valid = {
            triggered_desc = {
                trigger = { global_var:active_daedric_prince = "mehrunes_dagon_template" }
                desc = daedric_invasion.001.desc_dagon
            }
            triggered_desc = {
                trigger = { global_var:active_daedric_prince = "molag_bal_template" }
                desc = daedric_invasion.001.desc_molag
            }
            triggered_desc = {
                trigger = { global_var:active_daedric_prince = "boethiah_template" }
                desc = daedric_invasion.001.desc_boethiah
            }
            triggered_desc = {
                trigger = { global_var:active_daedric_prince = "hircine_template" }
                desc = daedric_invasion.001.desc_hircine
            }
            triggered_desc = {
                trigger = { global_var:active_daedric_prince = "namira_template" }
                desc = daedric_invasion.001.desc_namira
            }
            desc = daedric_invasion.001.desc_generic
        }
    }
    theme = war
    left_portrait = {
        character = scope:daedric_prince_char
        animation = personality_bold
    }

    option = {
        name = daedric_invasion.001.a  # "We must stand against this threat!"
        add_stress = -10
        add_prestige = 50
    }
    option = {
        name = daedric_invasion.001.b  # "Perhaps we can negotiate..."
        add_stress = 10
    }
}

# ------------------------------------------------------------
# daedric_invasion.045
# Player attempts to banish the Daedric Prince — dramatic duel.
# ------------------------------------------------------------
daedric_invasion.045 = {
    type = character_event
    title = daedric_invasion.045.t
    desc = daedric_invasion.045.desc
    theme = war
    left_portrait = {
        character = root
        animation = personality_bold
    }
    right_portrait = {
        character = scope:daedric_prince_char
        animation = anger
    }

    immediate = {
        # Simulate a duel outcome using the character's martial + prowess
        set_local_variable = {
            name = banish_chance
            value = root.martial
        }
        change_local_variable = {
            name = banish_chance
            add  = root.prowess
        }
    }

    option = {
        name = daedric_invasion.045.a  # "Press the advantage!"
        trigger = { local_var:banish_chance >= 30 }
        # Success
        end_daedric_invasion = yes
        add_prestige = 5000
        add_piety    = 2000
        add_trait    = daedric_champion
    }

    option = {
        name = daedric_invasion.045.b  # "Fall back — the Prince is too powerful."
        # Failure — wounded but alive
        add_prestige = -500
        add_stress   = 25
        add_modifier = { modifier = wounded days = 730 }
    }
}

# ------------------------------------------------------------
# daedric_invasion.050
# Monthly tick: increment invasion timer; check escalation.
# ------------------------------------------------------------
daedric_invasion.050 = {
    type = world_event
    hidden = yes

    trigger = {
        is_daedric_invasion_active = yes
    }

    immediate = {
        change_global_variable = {
            name = daedric_invasion_started_days
            add  = 30
        }
        if = {
            limit = { daedric_invasion_can_escalate = yes }
            escalate_daedric_invasion = yes
        }
    }
}

# ------------------------------------------------------------
# daedric_invasion.060
# On-death cleanup: remove Daedric Champion / Vassal traits.
# ------------------------------------------------------------
daedric_invasion.060 = {
    type = character_event
    hidden = yes

    trigger = {
        OR = {
            has_trait = daedric_champion
            has_trait = daedric_vassal
        }
    }

    immediate = {
        if = { limit = { has_trait = daedric_champion } remove_trait = daedric_champion }
        if = { limit = { has_trait = daedric_vassal   } remove_trait = daedric_vassal   }
    }
}
