#  necromancer_third_way.rank1      — Seeking (Arkay defied, Mannimarco rejected)
#  necromancer_third_way.rank2      — Striving (death taxonomy, Ideal Masters refused)

#  necromancer_third_way.apotheosis — Apotheosis: Death Sovereign
#  necromancer_third_way.unmoored   — Failure: soul partially dissociated
#
#  LORE BASIS:
#  ───────────
#  Independent necromancers existed in Tamriel outside the Worm Cult
#  and the Ideal Masters' framework — this is confirmed by UESP and
#  the Order of the Black Worm lore pages. Such practitioners were
#  hunted by both Mannimarco (who viewed unsanctioned necromancy as
#  apostasy and competition) and mainstream society.
#  The Third Way rejects both established necromantic hierarchies:
#  — Worm Apotheosis (Path I): Mannimarco's road, requires cult doctrine
#  — Ideal Masters path: the Soul Cairn bargain, leads to dissolution
#  The Third Way practitioner studies death as a domain to be governed
#  on entirely original terms. The apex — death_sovereign — is not the
#  god of necromancy. It is something newer and stranger.
#  [SOURCE: CANON — UESP Lore:Mannimarco; UESP Lore:Order of the Black Worm;
#   UESP Lore:Ideal Masters; UESP Lore:Soul Cairn]
# =============================================================

    title = necromancer_third_way.intro.t
    desc  = necromancer_third_way.intro.desc
    theme = learning

    trigger = {
        is_on_necromancer_third_way_path = yes

        add_stress = 20

            modifier = ww_path_necromancer_third_way_active
            years = 100
        }
    }
}

necromancer_third_way.apotheosis = {
    type = character_event
    title = necromancer_third_way.apotheosis.t
    desc  = necromancer_third_way.apotheosis.desc
    theme = learning

    trigger = {
        has_character_flag = ww_necromancer_third_way_active
        var:ww_necromancer_third_way_rank >= 4
        NOT = { has_character_flag = ww_necromancer_third_way_completed }
        NOT = { has_character_flag = ww_third_way_unmoored }
    }

    immediate = {
        complete_path_necromancer_third_way = yes
    }

    option = {
        name = necromancer_third_way.apotheosis.a   # "I am sovereign over my own ending. No cult. No masters. My own terms."
        add_prestige = 500
        add_learning_xp = 400
        add_stress = -25
    }

    option = {
        name = necromancer_third_way.apotheosis.b   # "Something new has been added to the taxonomy of death. My name is in it."
        add_prestige = 450
        add_piety = -50   # Arkay's clergy take note
        add_dread = 10
    }
}

necromancer_third_way.unmoored = {
    type = character_event
    title = necromancer_third_way.unmoored.t
    desc  = necromancer_third_way.unmoored.desc
    theme = learning

    trigger = {
        has_character_flag = ww_third_way_unmoored
        has_character_flag = ww_necromancer_third_way_active
    }

    immediate = {
        remove_trait = walking_ways_seeker
        clear_all_ww_active_flags = yes
        remove_variable = ww_necromancer_third_way_rank
        remove_variable = ww_necromancer_third_way_milestones
        remove_character_flag = ww_third_way_unmoored
        add_trait = lich_essence   # stumbled into near-death survival as a side-effect
        add_character_modifier = {
            modifier = soul_unmoored
            years = 30
        }
    }

    option = {
        name = necromancer_third_way.unmoored.a   # "No framework. No precedent. Just the raw consequence of standing at the boundary."
        add_stress = 40
        add_learning_xp = 300
        # The failure taught them why a framework exists — they may now pursue worm_apotheosis
    }
}
