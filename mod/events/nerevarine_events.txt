# =============================================================
#  The Nerevarine System — Events
#  Namespace: nerevarine
#  File: mod/events/nerevarine_events.txt
#
#  LORE:
#  The Nerevarine path does not end cleanly in every timeline.
#  The vessel of Azura's prophecy may fall in battle before reaching
#  Red Mountain, allowing another to take up the mantle.  They may
#  stand at the Heart itself and choose betrayal over destiny.  A
#  rival tapping the Heart's power may murder the Nerevarine to keep
#  the Heart bound.  And a heart_fused ruler given enough time may
#  transcend the role of vessel entirely — becoming the Sharmat
#  Enthroned, the Blight as throne.
#
#  EVENT OVERVIEW:
#  ──────────────
#  Succession:
#    nerevarine.001  — world notification: the prophecy's vessel has fallen
#
#  §21 Race-Agnostic / Multi-Spirit / Shezarrine events (NEW):
#    nerevarine.300  — Outlander Nerevarine: non-Dunmer claimant receives
#                      Azura's acknowledgement + Ashlander recognition vision
#    nerevarine.310  — Vision of Past Incarnations: all new claimants see
#                      the souls of previous Nervarines in Azura's realm;
#                      shezarrine_vessel claimants gain an extra option
#    nerevarine.320  — Dual-Soul Convergence: fires when a nerevarine_marked
#                      character also bears shezarrine_vessel; acknowledges
#                      the dual-soul nature; Shor's throne is NOT affected here
#                      (throne only empties when Shezarrine enters Sovngarde)
#
#  §22 Multi-Pretender / Pre-cursor events (NEW):
#    nerevarine.330  — True Nerevarine Proclaimed (hidden): fires when a
#                      character achieves nerevarine_victor; notifies all
#                      other nerevarine_marked vessels to become pretenders
#    nerevarine.331  — Pretender's Reckoning: fires for each surviving
#                      nerevarine_marked vessel when the prophecy is fulfilled;
#                      they lose nerevarine_marked and gain nerevarine_pretender
#    nerevarine.340  — The Prophecy Stirs (hidden yearly pre-cursor): fires
#                      for eligible rulers when dagoth_ur_awakening_possible
#                      but BEFORE the full awakening; auto-marks multiple
#                      candidates as nerevarine_marked across all races
#
#  Cross-path interactions:
#    nerevarine.010  — Nerevarine encounters a Heart Scholar (yearly trigger)
#    nerevarine.011  — Scholar assists: shares Kagrenac knowledge
#    nerevarine.012  — Scholar opposes: blocks the Nerevarine's path
#    nerevarine.020  — Nerevarine encounters a Heart Fused ruler (yearly trigger)
#    nerevarine.021  — Fused ruler offers to share the Heart's power
#    nerevarine.022  — Fused ruler fights to prevent the sever
#
#  Assassination (Heart Scholar/Fused vs. Nerevarine):
#    nerevarine.030  — The confrontation: scholar/fused attacks Nerevarine
#
#  Betrayal events (fired by kagrenac.030 option B and decisions):
#    nerevarine.200  — world notification: the Nerevarine has turned to Dagoth
#    nerevarine.210  — Betraying Dagoth for Mephala: the silent knife
#    nerevarine.211  — Betraying Dagoth for Boethiah: the open challenge
#
#  Sharmat's Dream milestones (heart_fused_ascendant progression):
#    nerevarine.099  — hidden yearly: advance Sharmat milestone chain
#    nerevarine.100  — Milestone 1: The Dream Spreads
#    nerevarine.101  — Milestone 2: Corprus as Gift
#    nerevarine.102  — Milestone 3: The Ghost Fence Falls
#    nerevarine.103  — Milestone 4: The Nerevarine Approaches
#    nerevarine.104  — Milestone 5: The Heart Sings
#    nerevarine.105  — Milestone 6: Sharmat Enthroned (apex)
#
#  KEY FLAGS & VARIABLES:
#  ──────────────────────
#  has_global_flag = nerevarine_quest_incomplete
#                    — set when nerevarine_marked character dies
#                      before nerevarine_victor; cleared on new claim
#  global_var:false_nerevarine_count
#                    — integer tracking failed incarnations;
#                      Azura grows harder to approach at 3+
#  global_var:active_nerevarine_spirit_count     ← NEW §21

#                      incremented on claim, decremented on death

#                    — set ONLY when a Shezarrine physically enters Sovngarde

#                      Shezarrine convergence alone (nerevarine.320 no longer

#                      cleared on that Shezarrine's death

#                    — marks the vessel that acknowledged their dual Nerevarine/
#                      Shezarrine nature (nerevarine.320 option A).  Shor's

# ---------------------------------------------------------------
# nerevarine.001  The Prophecy's Vessel Has Fallen
# Fires for Dunmer rulers when a nerevarine_marked character dies
# before completing the prophecy (on_character_death hook).
# Informs all Dunmer rulers that the mantle is now available.
# ---------------------------------------------------------------
nerevarine.001 = {
    type = character_event
    title = nerevarine.001.t
    desc = nerevarine.001.desc
    theme = faith

    option = {
        name = nerevarine.001.a   # "Azura's chosen has fallen. The prophecy is incomplete."
        add_piety    = 50
        add_stress   = 10
        add_character_modifier = {
            modifier = nerevarine_succession_open
            days     = 1825    # 5-year window of spiritual urgency
        }
    }

    option = {
        name = nerevarine.001.b   # "This changes nothing. Dagoth Ur still awakens and the Heart still calls."
        trigger = {
            OR = {
                has_trait = azura_champion
                has_trait = nerevarine_marked
            }
        }
        add_piety    = 100
        add_stress   = 20
        custom_tooltip = nerevarine_claim_available_tt
    }

    option = {
        name = nerevarine.001.c   # "The prophecy may yet find a new vessel — but not me."
        trigger = {
            NOT = { has_trait = azura_champion }
        }
        add_piety = 30
    }
}

# nerevarine.010  The Nerevarine and the Scholar — First Contact
# Fires yearly for nerevarine_marked rulers when a heart_scholar_ascendant
# ruler exists in the same kingdom or nearby.  Two paths meet at the
# edge of the same mystery.
# ---------------------------------------------------------------

    type = character_event
    title = nerevarine.010.t

    theme = intrigue

        has_trait = nerevarine_marked

    option = {
        name = nerevarine.010.a   # "Approach the scholar. They hold knowledge of the Heart's workings."
        add_character_flag = { flag = scholar_encounter_recent days = 1825 }
        trigger_event = { id = nerevarine.011 days = 14 }

        name = nerevarine.010.b   # "Keep distance. Their borrowed divinity may corrupt the prophecy's purpose."
        add_character_flag = { flag = scholar_encounter_recent days = 1825 }

        add_stress = 10
    }

    option = {
        name = nerevarine.010.c   # "Challenge them — no one should hold the Heart's power except Azura's chosen."
        trigger = {
            martial >= 12
        }
        add_character_flag = { flag = scholar_encounter_recent days = 1825 }
        trigger_event = { id = nerevarine.012 days = 7 }
    }
}

# ---------------------------------------------------------------
# nerevarine.030  The Confrontation — Slay the Nerevarine
# Fired by the slay_the_nerevarine decision (nerevarine_decisions.txt).
# A heart_scholar_ascendant or heart_fused_ascendant ruler launches
# an assassination attempt on the active Nerevarine-marked character.
# ---------------------------------------------------------------
nerevarine.030 = {
    type = character_event
    title = nerevarine.030.t
    desc = nerevarine.030.desc
    theme = intrigue

    option = {
        name = nerevarine.030.a   # "Strike cleanly. The Heart must remain."
        trigger = {
            OR = {
                intrigue >= 14
                martial  >= 14
            }
        }
        # Find and kill the Nerevarine
        random_living_character = {
            limit = {
                has_trait = nerevarine_marked
                NOT = { has_trait = nerevarine_victor }
            }
            kill_character = yes
        }
        set_global_flag = nerevarine_quest_incomplete
        if = {
            limit = { NOT = { exists = global_var:false_nerevarine_count } }
            set_global_variable = { name = false_nerevarine_count value = 0 }
        }
        change_global_variable = { name = false_nerevarine_count add = 1 }
        set_character_flag = { flag = nerevarine_slain_by_player }
        add_character_modifier = {
            modifier = heart_anchor_modifier
            days = 3650    # 10 years of Heart remaining bound
        }
        add_prestige = 1000
        add_piety    = -200    # Azura's displeasure at killing her chosen
        add_stress   = 20
        custom_tooltip = nerevarine_slain_tt
    }

    option = {
        name = nerevarine.030.b   # "The attempt fails — they are stronger than anticipated."
        trigger = {
            martial  < 14
            intrigue < 14
        }
        add_health   = -1
        add_prestige = -200
        add_stress   = 30
        add_piety    = -100
    }
}

# ---------------------------------------------------------------
# nerevarine.099  Sharmat's Dream — Hidden Milestone Progression
# Fires yearly for heart_fused_ascendant rulers while Dagoth Ur is
# awakened.  Advances the sharmat_rank variable and triggers the
# appropriate milestone event.
# ---------------------------------------------------------------
nerevarine.099 = {
    type = character_event
    hidden = yes

    trigger = {
        has_trait = heart_fused_ascendant
        NOT = { has_trait = sharmat_ascendant }
        dagoth_ur_awakened = yes
    }

    immediate = {
        # Initialize tracking variables if first time
        if = {
            limit = { NOT = { exists = var:sharmat_rank } }
            set_variable = { name = sharmat_rank value = 0 }
        }
        if = {
            limit = { NOT = { exists = var:sharmat_fused_years } }
            set_variable = { name = sharmat_fused_years value = 0 }
        }
        change_variable = { name = sharmat_fused_years add = 1 }

        # Milestone 1: Dream Spreads (1+ year fused)
        if = {
            limit = {
                var:sharmat_rank = 0
                var:sharmat_fused_years >= 1
            }
            trigger_event = { id = nerevarine.100 }
        }
        # Milestone 2: Corprus as Gift (3+ years, sixth house vassal present)
        else_if = {
            limit = {
                var:sharmat_rank = 1
                var:sharmat_fused_years >= 3
                any_vassal = { has_trait = sixth_house_cultist }
            }
            trigger_event = { id = nerevarine.101 }
        }
        # Milestone 3: Ghost Fence Falls (5+ years)
        else_if = {
            limit = {
                var:sharmat_rank = 2
                var:sharmat_fused_years >= 5
            }
            trigger_event = { id = nerevarine.102 }
        }
        # Milestone 4: Nerevarine Approaches (any living nerevarine_marked or victor exists)
        else_if = {
            limit = {
                var:sharmat_rank = 3
                OR = {
                    any_living_character = { has_trait = nerevarine_marked }
                    any_living_character = { has_trait = nerevarine_victor }
                }
            }
            trigger_event = { id = nerevarine.103 }
        }
        # Milestone 5: Heart Sings (10+ years fused)
        else_if = {
            limit = {
                var:sharmat_rank = 4
                var:sharmat_fused_years >= 10
            }
            trigger_event = { id = nerevarine.104 }
        }
        # Milestone 6: Sharmat Enthroned (rank 5 — fires immediately after Heart Sings)
        else_if = {
            limit = { var:sharmat_rank = 5 }
            trigger_event = { id = nerevarine.105 days = 90 }
        }
    }
}

# ---------------------------------------------------------------
# nerevarine.320  Dual-Soul Convergence — Nerevar and Shor, One Flesh
# A character who is both the Nerevarine (nerevarine_marked) and a
# Shezarrine (shezarrine_vessel) undergoes a profound metaphysical
# convergence.  Both souls are reincarnations of dead gods:
# Nerevar was the greatest champion Resdayn ever knew, and his soul
# is beloved of Azura; Shor is the dead god in the ground, the heart
# torn from the world.  When these two lineages meet in a single body
# the world notices.
#
# Lore basis: Wulf (an avatar of Talos, himself a Shezarrine) appears
# to the Nerevarine in Morrowind, proving two Shezarrine-touched
# beings can coexist on Nirn simultaneously.  Talos-as-Wulf is a
# manifest avatar, not the full soul; but it demonstrates that the
# Shezarrine lineage is not a singular monopoly.  The Dragonborn of
# Skyrim shows the further corollary: while a Shezarrine lives on
# Nirn, Shor's throne in Aetherius stands empty.
# [SOURCE: TES III:Morrowind — Wulf NPC; UESP Lore:Shezarrine;
#  UESP Lore:Shor; TES V:Skyrim — Sovngarde empty throne]
#
# Mechanical result: sets nerevarine_shezarrine_convergence character flag
# (the dual-soul nature is acknowledged) and applies dual_soul_convergence_modifier.
# NOTE: Shor's throne empty flag (shors_throne_nerevarine) is NOT set here.
# The throne only becomes empty when the Shezarrine physically enters Sovngarde
# (shezarrine.sovngarde_entry), NOT merely because the convergence exists on Nirn.
# Fired by: nerevarine.310 option B
# ---------------------------------------------------------------
nerevarine.320 = {
    type = character_event
    title = nerevarine.320.t
    desc = nerevarine.320.desc
    theme = faith

    option = {
        name = nerevarine.320.a   # "Two souls, one purpose — I carry the heart and the war-cry both."
        set_character_flag = nerevarine_shezarrine_convergence
        # NOTE: Shor's throne is NOT made empty simply because this convergence
        # happens on Nirn.  Shor's throne only stands empty when a Shezarrine
        # physically enters Sovngarde (shezarrine.sovngarde_entry event).
        # [SOURCE: TES V:Skyrim — Dragonborn enters Sovngarde; UESP Lore:Shor]
        add_piety    = 500
        add_prestige = 600
        add_stress   = -20
        add_character_modifier = {
            modifier = dual_soul_convergence_modifier
            years = 99    # effectively permanent — cleared on death
        }
        custom_tooltip = nerevarine_dual_soul_tt
    }

    option = {
        name = nerevarine.320.b   # "The weight of two dead gods is more than I can bear.  I choose Nerevar alone."
        add_piety    = 100
        add_prestige = 200
        add_stress   = 10
        # Set a flag so the yearly pulse (nerevarine.321) does not
        # keep offering the convergence to a character who declined.
        set_character_flag = nerevarine_shezarrine_declined
        # Character remains shezarrine_vessel and nerevarine_marked
        # but does not trigger the convergence — lore-valid: they
        # have the dual soul but suppress Shor's aspect deliberately
        custom_tooltip = nerevarine_only_path_tt
    }
}

# ---------------------------------------------------------------
# nerevarine.340  The Prophecy Stirs (hidden yearly pre-cursor)
# Fires for eligible rulers when dagoth_ur_awakening_possible = yes
# but BEFORE Dagoth Ur is fully awake.  This represents the
# prophetic force of Azura beginning to seek vessels across all
# of Tamriel — not one chosen hero, but many potential Incarnates
# simultaneously stirred, as the lore suggests the prophecy did:
# "The Nerevarine will come as a stranger to these lands."
# Multiple characters of different races are marked as potential
# vessels.  Only the first to fulfil the prophecy (nerevarine_victor)
# will be the True Nerevarine; the rest become False Incarnates.
#
# Non-canonical races (Dremora, Dwemer, Hist-bound Argonians, etc.)
# may still receive the mark if they carry azura_champion — Azura's
# judgement is the only gate, not bloodline.
#
# Mean-time-to-happen is short so that several characters are marked
# within the first year of dagoth_ur_awakening_possible.
# Registered in: on_yearly_pulse random_events weight 3
# ---------------------------------------------------------------
nerevarine.340 = {
    type = character_event
    hidden = yes

    trigger = {
        dagoth_ur_awakening_possible = yes
        NOT = { dagoth_ur_awakened = yes }   # fires as PRE-CURSOR, before full awakening
        NOT = { has_global_flag = dagoth_ur_defeated }
        is_ruler = yes
        is_adult = yes
        NOT = { has_trait = nerevarine_marked }
        NOT = { has_trait = nerevarine_victor }
        NOT = { has_trait = nerevarine_forsaken }
        NOT = { has_trait = nerevarine_pretender }
        NOT = { has_character_flag = nerevarine_precursor_offered }
        # Primary gate: Azura's recognition (any race)
        OR = {
            has_trait = azura_champion
            # Dunmer with cultural/spiritual ties to the prophecy
            AND = {
                has_culture = dunmer_culture
                piety    >= 150
                learning >= 8
            }
        }
    }

    mean_time_to_happen = {
        months = 4    # Multiple vessels will be marked in the first months of the stirring
        modifier = {
            factor = 0.5
            has_culture = dunmer_culture    # Dunmer vessels are more quickly called
        }
        modifier = {
            factor = 0.75
            has_trait = azura_champion
        }
    }

    immediate = {
        set_character_flag = nerevarine_precursor_offered
        add_trait = nerevarine_marked
        if = {
            limit = { NOT = { exists = global_var:active_nerevarine_spirit_count } }
            set_global_variable = { name = active_nerevarine_spirit_count value = 0 }
        }
        change_global_variable = { name = active_nerevarine_spirit_count add = 1 }
        if = {
            limit = { NOT = { exists = global_var:false_nerevarine_count } }
            set_global_variable = { name = false_nerevarine_count value = 0 }
        }
        change_global_variable = { name = false_nerevarine_count add = 1 }
        # Clear any open quest flag — the prophecy is actively seeking
        if = {
            limit = { has_global_flag = nerevarine_quest_incomplete }
            remove_global_flag = nerevarine_quest_incomplete
        }
        add_character_modifier = {
            modifier = dagoth_ur_champion_modifier
            days = 3650
        }
        # Non-Dunmer claimants get the outlander revelation
        if = {
            limit = { NOT = { has_culture = dunmer_culture } }
            trigger_event = { id = nerevarine.300 days = 3 }
        }
        # All new claimants see the vision of past incarnations
        trigger_event = { id = nerevarine.310 days = 7 }
    }

    option = { name = nerevarine.340.a }
}
