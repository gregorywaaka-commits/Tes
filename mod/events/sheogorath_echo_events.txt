#  creative chaos, the logic that reason cannot follow — WITHOUT claiming
#  the throne. They are the echo, not the source. Madness as insight.
#  The distinction is critical: Sheogorath's Echo closes the throne door.
#  A character who walks the echo path has chosen understanding over
#  succession, and the throne does not re-offer.
#  [SOFT CANON — extrapolated from mantling metaphysics; the partial
#   mantling / resonance concept is not explicitly stated in TES canon

sheogorath_echo.intro = {
    type = character_event
    title = sheogorath_echo.intro.t
    desc  = sheogorath_echo.intro.desc
    theme = intrigue

    option = {
        name = sheogorath_echo.intro.a   # "I understand what madness is. I will not become it."
        add_stress = -10
        add_piety = 20
        add_character_modifier = {
            modifier = ww_path_sheogorath_echo_active
            years = 100
        }
    }
}

sheogorath_echo.apotheosis = {
    type = character_event
    title = sheogorath_echo.apotheosis.t
    desc  = sheogorath_echo.apotheosis.desc
    theme = intrigue

    trigger = {
        has_character_flag = ww_sheogorath_echo_active
        var:ww_sheogorath_echo_rank >= 4
        NOT = { has_character_flag = ww_sheogorath_echo_completed }
        NOT = { has_character_flag = ww_sheogorath_echo_possessed }
    }

    immediate = {
        complete_path_sheogorath_echo = yes
    }

    option = {
        name = sheogorath_echo.apotheosis.a   # "I understand the Prince of Madness. I chose not to become him. The echo is enough."
        add_prestige = 350
        add_stress = -30
        add_piety = 100
    }

    option = {
        name = sheogorath_echo.apotheosis.b   # "The madness is a tool. I am still the hand that holds it."
        add_prestige = 300
        add_intrigue_xp = 300
    }
}

sheogorath_echo.possessed = {
    type = character_event
    title = sheogorath_echo.possessed.t
    desc  = sheogorath_echo.possessed.desc
    theme = intrigue

    trigger = {
        has_character_flag = ww_sheogorath_echo_possessed
        has_character_flag = ww_sheogorath_echo_active
    }

    immediate = {
        remove_trait = walking_ways_seeker
        clear_all_ww_active_flags = yes
        remove_variable = ww_sheogorath_echo_rank
        remove_variable = ww_sheogorath_echo_milestones
        remove_character_flag = ww_sheogorath_echo_possessed
        add_trait = lunatic
        add_trait = possessed
        add_character_modifier = {
            modifier = sheogorath_consumed
            years = 30
        }
    }

    option = {
        name = sheogorath_echo.possessed.a   # "I stopped echoing. Something walked in."
        add_stress = 50
        add_intrigue_xp = 150
        # Still some power — the possession brings its own gifts
    }
}
