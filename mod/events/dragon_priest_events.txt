# =============================================================
#  Dragon Priest / Alduin / Dragon Cult Events
#  Namespace: dragon_priest
#
#  Three interconnected systems:
#
#  SYSTEM 1 — Dragon Priest Masks (dragon_priest.000–005)
#    The eight Dragon Priests of the Merethic Era still haunt
#    their barrow tombs.  Their masks hold real power — and
#    real cost.  The Dragonborn and Voice-trained rulers are
#    most likely to sense their presence and hunt them.
#
#  SYSTEM 2 — Alduin the World-Eater (dragon_priest.010–012)
#    The firstborn of Akatosh.  He was cast into the timestream
#    at the close of the Merethic Era by the ancient Nord heroes
#    using Dragonrend.  His shadow crosses the dreams of the
#    Dragonborn and the most powerful Voice-users even now.
#
#  SYSTEM 3 — Dragon Cults (dragon_priest.020–022)
#    Before the Alduin rebellion was broken, mortals worshipped
#    dragons as gods.  These cults persist in hidden corners of
#    Nordic lands, preserving corrupted fragments of the Thu'um
#    and the old devotion to dragon-kind.
#
#  EK2 is required.  'dragonborn' is EK2's trait.
#  All traits and modifiers are defined in lore_races files.
#  Lore timeframe: 2E 440–896 (the Interregnum)
#
#  TRACKING:
#  ─────────
#  • dragon_masks_count   — character variable; total masks held
#  • dragon_priest_hunt_active  — currently pursuing a tomb (timed)
#  • dragon_priest_tomb_marked  — a tomb was located but not yet entered
#  • dragon_priest_stir_seen    — dragon_priest.004 already fired
#  • dragon_priest_hunted_3     — flag set when third mask is taken
#  • alduin_vision_seen         — dragon_priest.010 already fired
#  • alduin_shadow_seen         — dragon_priest.011 already fired
#  • alduin_aftermath_seen      — dragon_priest.012 already fired
#  • dragon_cult_member         — is an active dragon cult member
#  • dragon_cult_exposed        — cult has been exposed/renounced
# =============================================================

namespace = dragon_priest

# ---------------------------------------------------------------
# dragon_priest.000  —  The Barrow Disturbed
# Yearly pulse event.  A Dragon Priest's tomb stirs — felt only
# by those attuned to the Voice or the blood of a Dovahkiin.
# The ancient undead inside have not truly rested since the
# Merethic Era.  Something has woken them.  Something always does.
# Trigger conditions set in lore_races_on_actions.txt
# ---------------------------------------------------------------
dragon_priest.000 = {
    type  = character_event
    title = dragon_priest.000.t
    desc  = dragon_priest.000.desc
    theme = realm

    trigger = {
        is_ruler = yes
        is_adult = yes
        OR = {
            has_trait = dragonborn
            has_trait = thu_um_master
            has_trait = thu_um_adept
            has_trait = thu_um_initiate
            has_trait = greybeard_aspirant
            has_trait = greybeard_of_high_hrothgar
            has_trait = first_voice_of_the_greybeards
        }
        NOT = { has_character_flag = dragon_priest_hunt_active }
        NOT = { has_character_flag = dragon_priest_tomb_marked }
        NOT = { has_trait = dragon_priest_master }
    }

    option = {
        name = dragon_priest.000.a   # "Investigate. I can feel the power from here."
        trigger = { gold >= 100 }
        add_gold = -100
        add_character_flag = { flag = dragon_priest_hunt_active days = 365 }
        trigger_event = { id = dragon_priest.001 days = 30 }
        add_prestige = -50
        add_stress   = 10
    }

    option = {
        name = dragon_priest.000.b   # "Mark the barrow's location. I will return when prepared."
        add_character_flag = dragon_priest_tomb_marked
        add_stress = 5
    }

    option = {
        name = dragon_priest.000.c   # "Let the dead lie. There is enough trouble among the living."
        add_stress = 10
        add_piety  = 50   # The wise know when to leave ancient things undisturbed
    }
}

# ---------------------------------------------------------------
# dragon_priest.001  —  Into the Tomb
# Fired by hunt_dragon_priest decision (or dragon_priest.000 option A).
# The ruler has descended into a Dragon Priest's barrow.
# The air is wrong.  The walls are covered in carved dragon script.
# Something behind the final door has been awake for a very long time.
# Four paths determine the outcome — and the mask earned.
# ---------------------------------------------------------------
dragon_priest.001 = {
    type  = character_event
    title = dragon_priest.001.t
    desc  = dragon_priest.001.desc
    theme = war

    trigger = {
        has_character_flag = dragon_priest_hunt_active
        NOT = { has_character_flag = dragon_priest_tomb_cleared }
    }

    option = {
        # Overwhelm with Voice — speaks Dragonrend's conceptual opposite
        # into the tomb and unmakes the Dragon Priest's immortality.
        name = dragon_priest.001.a   # "I speak the Voice into the dark. Let the ancient one hear what has come for it."
        trigger = {
            OR = {
                has_trait = dragonborn
                has_trait = thu_um_master
                has_trait = greybeard_of_high_hrothgar
                has_trait = first_voice_of_the_greybeards
            }
            martial >= 14
        }
        add_character_flag = dragon_priest_tomb_cleared
        remove_character_flag = dragon_priest_hunt_active
        add_prestige = 500
        add_stress   = -10
        trigger_event = { id = dragon_priest.002 days = 7 }
    }

    option = {
        # Dispel its power — scholarly unravelling of the undead priest's
        # tether to the material world.  Requires deep understanding of
        # the magical principles that underpin Dragon Priest lichdom.
        name = dragon_priest.001.b   # "These bindings are ancient tonal architecture. I know how to unmake them."
        trigger = { learning >= 16 }
        add_character_flag = dragon_priest_tomb_cleared
        remove_character_flag = dragon_priest_hunt_active
        add_prestige = 400
        add_stress   = 5
        trigger_event = { id = dragon_priest.003 days = 7 }
    }

    option = {
        # Brute force — martial power and determination carry the ruler
        # through a brutal physical confrontation.  The priest is defeated
        # but not fully unmade.  A lesser prize is seized from the dust.
        name = dragon_priest.001.c   # "I have fought worse than animated corpses. I press through with steel and will."
        trigger = { martial >= 12 }
        add_character_flag = dragon_priest_tomb_cleared
        remove_character_flag = dragon_priest_hunt_active
        add_prowess  = -1   # the fight costs something; even victors carry their wounds
        add_prestige = 200
        add_stress   = 15
        # Partial victory — Otar's mask, the Priest of Storms
        add_modifier = {
            modifier = dragon_priest_mask_otar
            days = 3650
        }
        # Increment mask count for Konahrik threshold
        if = {
            limit = { NOT = { has_variable = dragon_masks_count } }
            set_variable = { name = dragon_masks_count value = 1 }
        }
        if = {
            limit = { has_variable = dragon_masks_count }
            change_variable = { name = dragon_masks_count add = 1 }
        }
        if = {
            limit = { check_variable = { name = dragon_masks_count value >= 2 } }
            add_character_flag = dragon_priest_hunted_2
        }
        if = {
            limit = { check_variable = { name = dragon_masks_count value >= 3 } }
            add_character_flag = dragon_priest_hunted_3
        }
    }

    option = {
        # Retreat — the living should not descend into the barrows of things
        # that have outlasted entire civilisations.  There is no shame
        # in knowing when to leave.  There is, however, considerable stress.
        name = dragon_priest.001.d   # "Not today. The tomb is sealed again. We leave."
        remove_character_flag = dragon_priest_hunt_active
        remove_character_flag = dragon_priest_tomb_cleared
        add_stress   = 25
        add_prestige = -100   # Word gets around that one fled a barrow
    }
}

# ---------------------------------------------------------------
# dragon_priest.005  —  Konahrik, the Mask of the Warlord
# The eighth mask.  The golden one.  In the lore, Konahrik can
# only be earned if all eight Dragon Priests are present at once —
# it summons the Warlord, the guardian of the mask, and the fight
# is the price of wearing the ultimate prize.
# In this system: the ruler has hunted enough priests to know
# how they think.  Now, at the site of the ancient dragon cult's
# greatest ritual chamber, all the masks together form a key.
# ---------------------------------------------------------------
dragon_priest.005 = {
    type  = character_event
    title = dragon_priest.005.t
    desc  = dragon_priest.005.desc
    theme = realm

    trigger = {
        is_ruler = yes
        has_character_flag = dragon_priest_hunted_3
        NOT = { has_trait = dragon_priest_master }
        NOT = { has_character_flag = dragon_priest_konahrik_seen }
    }

    option = {
        name = dragon_priest.005.a   # "The masks together form the lock. I speak the last Word and the golden mask falls."
        trigger = {
            OR = {
                has_trait = dragonborn
                has_trait = thu_um_master
                has_trait = first_voice_of_the_greybeards
            }
        }
        add_character_flag = dragon_priest_konahrik_seen
        add_trait    = dragon_priest_master
        add_prestige = 1000
        add_piety    = -200   # The accumulation of Dragon Priest power is spiritually costly
        add_stress   = 30
    }

    option = {
        name = dragon_priest.005.b   # "The ritual is complete. The golden mask is mine — its price, already paid."
        trigger = {
            martial  >= 16
            learning >= 14
        }
        add_character_flag = dragon_priest_konahrik_seen
        add_trait    = dragon_priest_master
        add_prestige = 800
        add_piety    = -200
        add_stress   = 40   # The weight of Konahrik is heavier than all the others together
        add_prowess  = -1   # The ritual demanded its physical toll
    }

    option = {
        name = dragon_priest.005.c   # "This is enough. I take what I have. The golden mask will wait."
        add_character_flag = dragon_priest_konahrik_seen
        add_prestige = 300
        add_stress   = 20
        # No Konahrik — the moment passed, and the vault sealed again
    }
}

        add_stress   = 5
    }

    option = {
        # Ignore — the safest option; the invitation disappears
        name = dragon_priest.020.c   # "I did not receive this letter. It does not exist."
        add_character_flag = dragon_cult_invitation_seen

    }

# In a torchlit hall deep beneath a ruin the cult calls sacred,

# They carry power.  It is the power of worship, not mastery —

# difficult to feel.

dragon_priest.021 = {
    type  = character_event
    title = dragon_priest.021.t
    desc  = dragon_priest.021.desc
    theme = faith

    trigger = {
        has_character_flag = dragon_cult_member
        NOT = { has_character_flag = dragon_cult_ritual_seen }
    }

    option = {
        # Full participation — the devotion takes hold completely
        name = dragon_priest.021.a   # "I join the chant. The Words are mangled but their power is real."

        add_trait    = dragon_cultist_devotee
        add_modifier = {
            modifier = dragon_cult_blessing
            days = 730
        }
        add_piety    = -200   # Dragon worship is heresy of the deepest kind

        add_stress   = 20
    }

    option = {
        # Partial participation — watching but not fully joining the chant
        name = dragon_priest.021.b   # "I observe. I learn. But the words themselves — I do not speak them."
        add_character_flag = dragon_cult_ritual_seen
        add_modifier = {
            modifier = dragon_cult_blessing
            days = 365
        }
        add_learning = 1   # Witnessing the ritual has scholarly value
        add_piety    = -100
        add_stress   = 15
    }

# ---------------------------------------------------------------
# dragon_priest.022  —  The Greybeards Send Word
# A monk of High Hrothgar arrives at the ruler's court bearing
# no weapon and no formal title — only a letter written in the
# spare, precise hand of the Greybeard Order.  The letter does
# not accuse.  It asks.  But the question inside it is the
# kind that can only be answered one way.
# ---------------------------------------------------------------
dragon_priest.022 = {
    type  = character_event
    title = dragon_priest.022.t
    desc  = dragon_priest.022.desc
    theme = realm

    trigger = {
        is_ruler = yes
        has_character_flag = dragon_cult_member
        NOT = { has_character_flag = dragon_cult_greybeard_seen }
        OR = {
            has_trait = greybeard_aspirant
            has_character_flag = greybeard_quest_1_seen
            has_character_flag = greybeard_quest_2_seen
            has_character_flag = greybeard_quest_4_seen
        }
    }

    option = {
        # Renounce the cult — honesty, stress, loss of the cult's favour
        name = dragon_priest.022.a   # "The monk is right. I renounce it. Every part of it."
        add_character_flag = dragon_cult_greybeard_seen
        add_character_flag = dragon_cult_exposed
        remove_character_flag = dragon_cult_member
        add_piety    = 200
        add_prestige = -100   # Renouncing a path is public; the court notices
        add_stress   = 40
    }

    option = {
        # Deny — intrigue check; lie convincingly to the Greybeard monk
        name = dragon_priest.022.b   # "A misunderstanding. I have never attended any such gathering."
        add_character_flag = dragon_cult_greybeard_seen
        trigger = { intrigue >= 12 }
        add_piety    = -100   # The lie costs something even if it succeeds
        add_stress   = 30
        # The cult membership persists — and so does the danger
    }

    option = {
        # Fight — removing the evidence entirely; a poor choice
        name = dragon_priest.022.c   # "The monk will not leave this court with what they know."
        add_character_flag = dragon_cult_greybeard_seen
        add_piety    = -400   # Harming a Greybeard monk is beyond ordinary transgression
        add_prestige = -300
        add_stress   = 50
        add_dread    = 20
    }
}
