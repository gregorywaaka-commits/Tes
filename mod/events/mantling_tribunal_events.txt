#  their corresponding Good Daedra (Anticipation) mantle:
#    Mephala → Vivec     (secret murder becomes art-and-war)
#    Boethiah → Almalexia (treachery becomes protection)

mantling_tribunal.vivec_intro = {
    type = character_event
    title = mantling_tribunal.vivec_intro.t
    desc  = mantling_tribunal.vivec_intro.desc

    option = {
        name = mantling_tribunal.vivec_intro.a
        add_stress = 15
        add_piety = 50
        add_character_modifier = {
            modifier = ww_path_mantling_active
            years = 100
        }
    }
}

mantling_tribunal.vivec_rank3 = {
    type = character_event
    title = mantling_tribunal.vivec_rank3.t
    desc  = mantling_tribunal.vivec_rank3.desc
    theme = learning

    # Heart Temptation option — only if character does not already have heart_scholar_ascendant
    option = {
        name = mantling_tribunal.vivec_rank3.a   # "The contradiction holds. I do not need the Heart."
        trigger = { always = yes }
        add_prestige = 200
        add_piety = 150
        add_stress = -15
        set_character_flag = ww_mantling_vivec_milestone_apotheosis
    }

    option = {
        name = mantling_tribunal.vivec_rank3.heart_temptation   # "The Heart would make this complete. I feel its pull."
        trigger = {
            NOT = { has_trait = heart_scholar_ascendant }
        }
        add_prestige = 150
        add_piety = 100
        add_stress = 20
        # Sets a flag that the Heart has been acknowledged — may unlock Path F branching
        set_character_flag = ww_tribunal_heart_temptation_vivec
    }
}

mantling_tribunal.vivec_apotheosis = {
    type = character_event
    title = mantling_tribunal.vivec_apotheosis.t
    desc  = mantling_tribunal.vivec_apotheosis.desc
    theme = learning

    trigger = {
        has_character_flag = ww_mantling_vivec_active
        var:ww_mantling_vivec_rank >= 4
        NOT = { has_character_flag = ww_mantling_vivec_completed }
    }

    immediate = {
        complete_path_mantling_vivec = yes
    }

    option = {
        name = mantling_tribunal.vivec_apotheosis.a   # "Warrior and poet. Both. Neither. The pattern is complete."
        add_prestige = 400
        add_piety = 300
    }

    option = {
        name = mantling_tribunal.vivec_apotheosis.b   # "The 36 Lessons were not autobiography. They were instructions."
        add_prestige = 350
        add_piety = 250
        add_learning = 1
    }
}
