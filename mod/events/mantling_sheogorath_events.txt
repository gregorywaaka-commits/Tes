mantling_sheogorath.apotheosis = {
    type = character_event
    title = mantling_sheogorath.apotheosis.t
    desc  = mantling_sheogorath.apotheosis.desc
    theme = intrigue

    trigger = {
        has_character_flag = ww_mantling_sheogorath_active
        var:ww_mantling_sheogorath_rank >= 4
        NOT = { has_character_flag = ww_mantling_sheogorath_completed }
        NOT = { has_character_flag = ww_daedric_mantle_failed }
    }

    immediate = {
        complete_path_mantling_sheogorath = yes
    }

    option = {
        name = mantling_sheogorath.apotheosis.a   # "The cheese has been found. The realm is mine."
        add_prestige = 400
        add_piety = 150
        add_dread = 20
    }

    option = {
        name = mantling_sheogorath.apotheosis.b   # "I am the new Prince. I am also still myself."
        add_prestige = 350
        add_stress = -30
    }
}

mantling_sheogorath.possession = {
    type = character_event
    title = mantling_sheogorath.possession.t
    desc  = mantling_sheogorath.possession.desc
    theme = intrigue

    trigger = {
        has_character_flag = ww_daedric_mantle_failed
        has_character_flag = ww_mantling_sheogorath_active
    }

    immediate = {
        # The Grey March claims the mantler — deep madness, not apotheosis
        remove_trait = walking_ways_seeker
        clear_all_ww_active_flags = yes
        remove_variable = ww_mantling_sheogorath_rank
        remove_variable = ww_mantling_sheogorath_milestones
        remove_character_flag = ww_daedric_mantle_failed
        add_trait = lunatic
        add_trait = possessed
    }

    option = {
        name = mantling_sheogorath.possession.a   # "Not Sheogorath. Not myself. Something in between."
        add_stress = 40
        add_prestige = 75
        # Still some power, but not the principled madness sought
    }
}
